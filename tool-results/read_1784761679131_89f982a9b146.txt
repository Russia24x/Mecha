     1→/**
     2→ * MECHA: LAST PROTOCOL — Area Loader
     3→ * Converts AreaData into Matter physics bodies + visual rectangles + triggers.
     4→ * Reads platform/hazard data from AreaData.sections.
     5→ * Independent of entities — just builds the physical world.
     6→ */
     7→import Phaser from 'phaser';
     8→import { GAME, COLORS } from '../shared/Constants';
     9→import { PhysicsSystem } from '../systems/PhysicsSystem';
    10→import { WorldSystem } from './WorldSystem';
    11→import { t } from '../systems/LocalizationSystem';
    12→import type { AreaData, SectionData, PlatformData, HazardData, LoreObjectData, LandmarkData } from '../data/types';
    13→
    14→export interface LoadedArea {
    15→  solids: Phaser.Physics.Matter.Image[];
    16→  sectionTriggers: Phaser.Physics.Matter.Image[];
    17→  checkpointTriggers: Phaser.Physics.Matter.Image[];
    18→  bossEntryTrigger: Phaser.Physics.Matter.Image | null;
    19→  hazardTriggers: Phaser.Physics.Matter.Image[];
    20→  visualRects: Phaser.GameObjects.Rectangle[];
    21→  loreObjects: Phaser.GameObjects.Container[];
    22→  landmarks: Phaser.GameObjects.Container[];
    23→  grappleAnchors: Phaser.GameObjects.Container[];
    24→  empDoors: Phaser.GameObjects.Container[];
    25→  shortcuts: Phaser.GameObjects.Container[];
    26→  collectibles: Phaser.GameObjects.Container[];
    27→}
    28→
    29→export class AreaLoader {
    30→  private scene: Phaser.Scene;
    31→  private physics: PhysicsSystem;
    32→  private regionId: string = 'factory';
    33→  private activeTweens: Phaser.Tweens.Tween[] = [];
    34→
    35→  constructor(scene: Phaser.Scene, physics: PhysicsSystem) {
    36→    this.scene = scene;
    37→    this.physics = physics;
    38→  }
    39→
    40→  /** Create a tween and track it for cleanup on unload. */
    41→  private trackedTween(config: Phaser.Types.Tweens.TweenBuilderConfig): Phaser.Tweens.Tween {
    42→    const t = this.scene.tweens.add(config);
    43→    this.activeTweens.push(t);
    44→    return t;
    45→  }
    46→
    47→  /**
    48→   * Build the full area: floor, ceiling, per-section platforms, hazards,
    49→   * section triggers, checkpoint triggers, boss entry trigger.
    50→   * RegionId controls ALL visual styling — factory vs forest are completely separate.
    51→   */
    52→  load(area: AreaData): LoadedArea {
    53→    this.regionId = area.regionId;  // ── Store region for all visual decisions
    54→    const result: LoadedArea = {
    55→      solids: [],
    56→      sectionTriggers: [],
    57→      checkpointTriggers: [],
    58→      bossEntryTrigger: null,
    59→      hazardTriggers: [],
    60→      visualRects: [],
    61→      loreObjects: [],
    62→      landmarks: [],
    63→      grappleAnchors: [],
    64→      empDoors: [],
    65→      shortcuts: [],
    66→      collectibles: [],
    67→    };
    68→
    69→    // Floor (spans entire area)
    70→    this.addSolid(result, area.totalWidth / 2, GAME.HEIGHT, area.totalWidth, 80);
    71→    // Ceiling
    72→    this.addSolid(result, area.totalWidth / 2, -20, area.totalWidth, 40);
    73→
    74→    // Per-section platforms + hazards
    75→    for (const section of area.sections) {
    76→      this.buildSection(result, section, area.sectionWidth);
    77→    }
    78→
    79→    // Section triggers (at start of each section)
    80→    for (let i = 0; i < area.sections.length; i++) {
    81→      const sec = area.sections[i];
    82→      const trigger = this.physics.addSensor(sec.x + 80, GAME.HEIGHT / 2, 40, GAME.HEIGHT, `section-${sec.id}`);
    83→      trigger.setData('sectionId', sec.id);
    84→      result.sectionTriggers.push(trigger);
    85→    }
    86→
    87→    // Checkpoint triggers
    88→    for (const secId of area.checkpointSections) {
    89→      const cpX = (secId - 1) * area.sectionWidth + 640;
    90→      const cp = this.physics.addSensor(cpX, GAME.HEIGHT - 80, 60, 60, `checkpoint-${secId}`);
    91→      cp.setData('isCheckpoint', true);
    92→      cp.setData('checkpointSection', secId);
    93→      result.checkpointTriggers.push(cp);
    94→    }
    95→
    96→    // Boss entry trigger (section 6 = last section with bossId)
    97→    const bossSection = area.sections.find(s => s.bossId);
    98→    if (bossSection) {
    99→      const bossX = bossSection.x + 400;
   100→      const trigger = this.physics.addSensor(bossX, GAME.HEIGHT - 100, 40, 200, 'boss-entry');
   101→      trigger.setData('isBossEntry', true);
   102→      trigger.setData('bossId', bossSection.bossId);
   103→      result.bossEntryTrigger = trigger;
   104→    }
   105→
   106→    return result;
   107→  }
   108→
   109→  private buildSection(result: LoadedArea, section: SectionData, sectionWidth: number): void {
   110→    if (section.platforms) {
   111→      for (const plat of section.platforms) {
   112→        this.addSolid(result, plat.x, plat.y, plat.w, plat.h);
   113→      }
   114→    }
   115→    // Hazards
   116→    if (section.hazards) {
   117→      for (const hazard of section.hazards) {
   118→        const trigger = this.physics.addSensor(hazard.x, hazard.y, hazard.w, hazard.h, `hazard-${hazard.type}`);
   119→        trigger.setData('isHazard', true);
   120→        trigger.setData('hazardDamage', hazard.damage);
   121→        result.hazardTriggers.push(trigger);
   122→        // ── Graphical hazard visuals per type ──
   123→        const hazardVis = this.createHazardVisual(hazard);
   124→        result.visualRects.push(hazardVis as unknown as Phaser.GameObjects.Rectangle);
   125→      }
   126→    }
   127→    // Lore Objects — interactable environmental storytelling
   128→    if (section.loreObjects) {
   129→      for (const lore of section.loreObjects) {
   130→        const container = this.createLoreObject(lore);
   131→        result.loreObjects.push(container);
   132→      }
   133→    }
   134→    // Landmarks — large visual structures
   135→    if (section.landmarks) {
   136→      for (const lm of section.landmarks) {
   137→        const container = this.createLandmark(lm);
   138→        result.landmarks.push(container);
   139→      }
   140→    }
   141→    // Grapple Anchors — ability-gated (Metroidvania)
   142→    if (section.grappleAnchors) {
   143→      for (const anchor of section.grappleAnchors) {
   144→        const container = this.createGrappleAnchor(anchor);
   145→        result.grappleAnchors.push(container);
   146→      }
   147→    }
   148→    // EMP Doors — ability-gated (Metroidvania)
   149→    if (section.empDoors) {
   150→      for (const door of section.empDoors) {
   151→        const container = this.createEmpDoor(door);
   152→        result.empDoors.push(container);
   153→      }
   154→    }
   155→    // Shortcuts — one-way doors (Dark Souls style backtracking)
   156→    if (section.shortcuts) {
   157→      for (const sc of section.shortcuts) {
   158→        const container = this.createShortcut(sc);
   159→        result.shortcuts.push(container);
   160→      }
   161→    }
   162→    // Collectibles — pickups (health/energy fragments, skill points)
   163→    if (section.collectibles) {
   164→      for (const col of section.collectibles) {
   165→        const container = this.createCollectible(col);
   166→        result.collectibles.push(container);
   167→      }
   168→    }
   169→  }

   171→  /** Create a grapple anchor — glowing ring that the player can grapple to. */
   172→  private createGrappleAnchor(anchor: { id: string; x: number; y: number }): Phaser.GameObjects.Container {
   173→    const container = this.scene.add.container(anchor.x, anchor.y);
   174→    // Outer glow ring (pulsing cyan)
   175→    const glow = this.scene.add.circle(0, 0, 14, 0x66f0ff, 0.15);
   176→    glow.setBlendMode(Phaser.BlendModes.ADD);
   177→    container.add(glow);
   178→    this.trackedTween({
   179→      targets: glow, alpha: { from: 0.08, to: 0.25 }, scale: { from: 0.9, to: 1.2 },
   180→      duration: 1200, yoyo: true, repeat: -1, ease: 'Sine.inOut',
   181→    });
   182→    // Inner ring (solid)
   183→    const ring = this.scene.add.circle(0, 0, 8, 0x66f0ff, 0);
   184→    ring.setStrokeStyle(2, 0x66f0ff, 0.8);
   185→    container.add(ring);
   186→    // Center dot
   187→    const dot = this.scene.add.circle(0, 0, 2, 0x66f0ff, 0.9);
   188→    dot.setBlendMode(Phaser.BlendModes.ADD);
   189→    container.add(dot);
   190→    // Rotating spokes (4 small lines)
   191→    for (let i = 0; i < 4; i++) {
   192→      const angle = (i / 4) * Math.PI * 2;
   193→      const spoke = this.scene.add.rectangle(
   194→        Math.cos(angle) * 6, Math.sin(angle) * 6, 4, 1, 0x66f0ff, 0.6
   195→      );
   196→      spoke.setRotation(angle);
   197→      spoke.setBlendMode(Phaser.BlendModes.ADD);
   198→      container.add(spoke);
   199→    }
   200→    // Slow rotation
   201→    this.trackedTween({ targets: container, rotation: Math.PI * 2, duration: 4000, repeat: -1, ease: 'Linear' });
   202→    container.setDepth(6);
   203→    container.setData('grappleAnchorId', anchor.id);
   204→    container.setData('isGrappleAnchor', true);
   205→    return container;
   206→  }

   208→  /** Create an EMP door — locked barrier that opens when hit by EMP pulse. */
   209→  private createEmpDoor(door: { id: string; x: number; y: number; w: number; h: number }): Phaser.GameObjects.Container {
   210→    const container = this.scene.add.container(door.x, door.y);
   211→    // Door body (magenta barrier)
   212→    const body = this.scene.add.rectangle(0, 0, door.w, door.h, 0x602080, 0.6);
   213→    body.setStrokeStyle(2, 0xc060ff, 0.8);
   214→    container.add(body);
   215→    // Energy field lines (horizontal, animated)
   216→    for (let i = 0; i < 3; i++) {
   217→      const line = this.scene.add.rectangle(0, -door.h / 4 + i * door.h / 4, door.w - 4, 1, 0xc060ff, 0.5);
   218→      line.setBlendMode(Phaser.BlendModes.ADD);
   219→      container.add(line);
   220→      this.trackedTween({
   221→        targets: line, alpha: { from: 0.2, to: 0.7 }, duration: 600 + i * 200, yoyo: true, repeat: -1,
   222→      });
   223→    }
   224→    // Lock indicator (small icon)
   225→    const lock = this.scene.add.text(0, 0, '🔒', { fontSize: '10px', color: '#c060ff' }).setOrigin(0.5);
   226→    container.add(lock);
   227→    container.setDepth(7);
   228→    container.setData('empDoorId', door.id);
   229→    container.setData('isEmpDoor', true);
   230→    container.setData('empDoorOpen', false);
   231→    container.setSize(door.w, door.h);
   232→    // ── FIX Bug 1: Add physics body so the door actually blocks the player ──
   233→    const physicsBody = this.physics.addStaticRect(door.x, door.y, door.w, door.h);
   234→    physicsBody.setData('isEmpDoorBody', true);
   235→    physicsBody.setData('empDoorId', door.id);
   236→    container.setData('physicsBody', physicsBody);
   237→    return container;
   238→  }

   240→  /** Create a shortcut door — one-way, opens from one side and stays open. */
   241→  private createShortcut(sc: { id: string; x: number; y: number; w: number; h: number; toSection: number; opensFrom: string }): Phaser.GameObjects.Container {
   242→    const container = this.scene.add.container(sc.x, sc.y);
   243→    // Door frame (amber, industrial)
   244→    const frame = this.scene.add.rectangle(0, 0, sc.w + 8, sc.h + 8, 0x1a1814, 0.9);
   245→    frame.setStrokeStyle(2, 0xffc040, 0.6);
   246→    container.add(frame);
   247→    // Door body (closed = solid, open = transparent)
   248→    const body = this.scene.add.rectangle(0, 0, sc.w, sc.h, 0x2a2018, 0.85);
   249→    body.setStrokeStyle(1, 0xffc040, 0.4);
   250→    container.add(body);
   251→    // "Opens from" indicator — small arrow on the correct side
   252→    const arrowColor = 0xffc040;
   253→    let arrowX = 0, arrowY = 0, arrowRot = 0;
   254→    switch (sc.opensFrom) {
   255→      case 'left':  arrowX = -sc.w / 2 - 8; arrowRot = Math.PI; break;
   256→      case 'right': arrowX = sc.w / 2 + 8;  arrowRot = 0; break;
   257→      case 'top':   arrowY = -sc.h / 2 - 8; arrowRot = Math.PI / 2; break;
   258→      case 'bottom': arrowY = sc.h / 2 + 8; arrowRot = -Math.PI / 2; break;
   259→    }
   260→    const arrow = this.scene.add.triangle(arrowX, arrowY, -4, -4, 4, -4, 0, 4, arrowColor, 0.8);
   261→    arrow.setRotation(arrowRot);
   262→    container.add(arrow);
   263→    // Label
   264→    const label = this.scene.add.text(0, 0, '⇌', { fontSize: '12px', color: '#ffc040' }).setOrigin(0.5);
   265→    container.add(label);
   266→
   267→    container.setDepth(6);
   268→    container.setData('shortcutId', sc.id);
   269→    container.setData('isShortcut', true);
   270→    container.setData('shortcutOpen', false);
   271→    container.setData('opensFrom', sc.opensFrom);
   272→    container.setData('toSection', sc.toSection);
   273→    container.setSize(sc.w, sc.h);
   274→    // ── FIX Bug 1: Add physics body so the shortcut door actually blocks the player ──
   275→    const physicsBody = this.physics.addStaticRect(sc.x, sc.y, sc.w, sc.h);
   276→    physicsBody.setData('isShortcutBody', true);
   277→    physicsBody.setData('shortcutId', sc.id);
   278→    container.setData('physicsBody', physicsBody);
   279→    return container;
   280→  }

   282→  /** Create a collectible pickup — glowing orb that grants a reward. */
   283→  private createCollectible(col: { id: string; type: string; x: number; y: number; requiredAbility?: string }): Phaser.GameObjects.Container {
   284→    const container = this.scene.add.container(col.x, col.y);
   285→    // Color per type
   286→    const colors: Record<string, number> = {
   287→      health_fragment: 0x40d070,
   288→      energy_fragment: 0x4090ff,
   289→      skill_point: 0xffc040,
   290→      weapon_part: 0xff80ff,
   291→    };
   292→    const color = colors[col.type] ?? 0xffffff;
   293→
   294→    // Outer glow (pulsing)
   295→    const glow = this.scene.add.circle(0, 0, 16, color, 0.15);
   296→    glow.setBlendMode(Phaser.BlendModes.ADD);
   297→    container.add(glow);
   298→    this.trackedTween({
   299→      targets: glow, alpha: { from: 0.08, to: 0.25 }, scale: { from: 0.9, to: 1.3 },
   300→      duration: 1000, yoyo: true, repeat: -1, ease: 'Sine.inOut',
   301→    });
   302→
   303→    // Core orb
   304→    const orb = this.scene.add.circle(0, 0, 5, color, 0.9);
   305→    orb.setBlendMode(Phaser.BlendModes.ADD);
   306→    container.add(orb);
   307→
   308→    // Type-specific icon (small shape inside)
   309→    let icon: Phaser.GameObjects.Shape;
   310→    if (col.type === 'health_fragment') {
   311→      // Plus sign
   312→      icon = this.scene.add.rectangle(0, 0, 6, 2, 0xffffff, 0.9);
   313→      container.add(icon);
   314→      const icon2 = this.scene.add.rectangle(0, 0, 2, 6, 0xffffff, 0.9);
   315→      container.add(icon2);
   316→    } else if (col.type === 'energy_fragment') {
   317→      // Diamond
   318→      icon = this.scene.add.polygon(0, 0, [0, -4, 4, 0, 0, 4, -4, 0], 0xffffff, 0.9);
   319→      container.add(icon);
   320→    } else if (col.type === 'skill_point') {
   321→      // Star (approximated with triangle)
   322→      icon = this.scene.add.triangle(0, 0, -4, 3, 4, 3, 0, -4, 0xffffff, 0.9);
   323→      container.add(icon);
   324→    } else {
   325→      // weapon_part — small square
   326→      icon = this.scene.add.rectangle(0, 0, 4, 4, 0xffffff, 0.9);
   327→      container.add(icon);
   328→    }
   329→
   330→    // Float animation
   331→    this.trackedTween({
   332→      targets: container, y: col.y - 6, duration: 1500, yoyo: true, repeat: -1, ease: 'Sine.inOut',
   333→    });
   334→    // Slow rotation on the glow
   335→    this.trackedTween({
   336→      targets: glow, rotation: Math.PI * 2, duration: 3000, repeat: -1, ease: 'Linear',
   337→    });
   338→
   339→    container.setDepth(8);
   340→    container.setData('collectibleId', col.id);
   341→    container.setData('isCollectible', true);
   342→    container.setData('collectibleType', col.type);
   343→    container.setData('collected', false);
   344→    container.setData('requiredAbility', col.requiredAbility || null);  // ── FIX Bug 5 ──
   345→    container.setSize(20, 20);
   346→    container.setInteractive({ useHandCursor: false });
   347→    return container;
   348→  }

   350→  private addSolid(result: LoadedArea, x: number, y: number, w: number, h: number): void {
   351→    const s = this.physics.addStaticRect(x, y, w, h);
   352→    const g = this.scene.add.graphics();
   353→    g.setDepth(5);
   354→
   355→    // ── Region-specific platform rendering ──
   356→    const isWall = h > 80 && w < 100;
   357→    const isFloor = h <= 30 && w >= 100;
   358→    const isLedge = h <= 30 && w < 100;
   359→    const isPillar = h > 80 && w >= 100;
   360→
   361→    if (this.regionId === 'forest') {
   362→      this.drawForestPlatform(g, w, h);
   363→    } else {
   364→      // Factory (default)
   365→      if (isWall) this.drawWall(g, w, h);
   366→      else if (isFloor) this.drawFloor(g, w, h);
   367→      else if (isLedge) this.drawLedge(g, w, h);
   368→      else if (isPillar) this.drawPillar(g, w, h);
   369→      else this.drawGeneric(g, w, h);
   370→    }
   371→
   372→    g.setPosition(x, y);
   373→    result.solids.push(s);
   374→    result.visualRects.push(g as unknown as Phaser.GameObjects.Rectangle);
   375→
   376→    // ── Region-specific depth decorations ──
   377→    if (this.regionId === 'factory') {
   378→      if (isFloor && w >= 120) this.addFloorDecorations(result, x, y, w, h);
   379→      if (isWall && h > 150) this.addWallDecorations(result, x, y, w, h);
   380→      if (isFloor && w >= 100) {
   381→        if (Math.random() < 0.25) this.addElectricalSparks(result, x + (Math.random() - 0.5) * w * 0.6, y + h / 2 + 6);
   382→        if (Math.random() < 0.15) this.addFireHazard(result, x + (Math.random() - 0.5) * w * 0.5, y + h / 2 - 2);
   383→        if (Math.random() < 0.20) this.addSteamVent(result, x + (Math.random() - 0.5) * w * 0.6, y - h / 2 - 4);
   384→      }
   385→      // ── Visual supports for floating platforms (Issue #4 fix) ──
   386→      // Add cables/pipes hanging DOWN from platform bottom to make it look
   387→      // supported rather than floating in mid-air.
   388→      if ((isFloor || isLedge) && y < GAME.HEIGHT - 100) {
   389→        this.addPlatformSupports(result, x, y, w, h);
   390→      }
   391→    } else if (this.regionId === 'forest') {
   392→      if (isFloor && w >= 100) this.addForestDecorations(result, x, y, w, h);
   393→      if (isWall && h > 100) this.addForestWallDecorations(result, x, y, w, h);
   394→    }
   395→  }

   397→  /**
   398→   * Add visual supports (hanging cables + thin pipes) below floating platforms.
   399→   * This makes platforms look anchored rather than floating in mid-air.
   400→   * Purely cosmetic — no physics body.
   401→   */
   402→  private addPlatformSupports(result: LoadedArea, x: number, y: number, w: number, h: number): void {
   403→    const g = this.scene.add.graphics();
   404→    g.setDepth(4);  // slightly behind platform (depth 5)
   405→    const bottomY = y + h / 2;
   406→
   407→    // Hanging cables from platform bottom — 2-3 thin vertical lines
   408→    const cableCount = w >= 200 ? 3 : 2;
   409→    for (let i = 0; i < cableCount; i++) {
   410→      const cx = x - w / 2 + (w / (cableCount + 1)) * (i + 1);
   411→      const cableLen = 20 + Math.random() * 40;
   412→      // Cable (dark thin line)
   413→      g.lineStyle(2, 0x1a1e28, 0.8);
   414→      g.beginPath();
   415→      g.moveTo(cx, bottomY);
   416→      g.lineTo(cx, bottomY + cableLen);
   417→      g.strokePath();
   418→      // Cable end cap (small circle)
   419→      g.fillStyle(0x2a3040, 0.7);
   420→      g.fillCircle(cx, bottomY + cableLen, 2);
   421→    }
   422→
   423→    // Thin support strut at each end (diagonal brace to wall or ground)
   424→    const strutLen = Math.min(60, (GAME.HEIGHT - 80) - bottomY);
   425→    if (strutLen > 20) {
   426→      const leftX = x - w / 2 + 4;
   427→      const rightX = x + w / 2 - 4;
   428→      // Left strut (diagonal going down-out)
   429→      g.lineStyle(3, 0x2a3040, 0.6);
   430→      g.beginPath();
   431→      g.moveTo(leftX, bottomY);
   432→      g.lineTo(leftX - 8, bottomY + strutLen);
   433→      g.strokePath();
   434→      // Right strut
   435→      g.beginPath();
   436→      g.moveTo(rightX, bottomY);
   437→      g.lineTo(rightX + 8, bottomY + strutLen);
   438→      g.strokePath();
   439→    }
   440→
   441→    g.setPosition(0, 0);
   442→    result.visualRects.push(g as unknown as Phaser.GameObjects.Rectangle);
   443→  }

   445→  /** Forest platform — mossy stone/wood, completely different from factory metal. */
   446→  private drawForestPlatform(g: Phaser.GameObjects.Graphics, w: number, h: number): void {
   447→    // Body — dark mossy stone
   448→    g.fillStyle(0x1a2a14, 1);
   449→    g.fillRoundedRect(-w / 2, -h / 2, w, h, 3);
   450→    // Top surface — moss layer (green)
   451→    g.fillStyle(0x2a4a20, 0.8);
   452→    g.fillRoundedRect(-w / 2, -h / 2, w, 4, 2);
   453→    // Moss highlights
   454→    g.fillStyle(0x4a6a30, 0.4);
   455→    for (let mx = -w / 2 + 6; mx < w / 2 - 4; mx += 10) {
   456→      g.fillCircle(mx, -h / 2 + 2, 1 + Math.random());

   777→  /** Electrical short — broken wire with arcing blue-white sparks. */
   778→  private addElectricalSparks(result: LoadedArea, x: number, y: number): void {
   779→    // Broken wire stub
   780→    const wire = this.scene.add.graphics();
   781→    wire.setDepth(6);
   782→    wire.lineStyle(2, 0x2a2820, 0.8);
   783→    wire.beginPath();
   784→    wire.moveTo(x, y - 8);
   785→    wire.lineTo(x + 2, y);
   786→    wire.strokePath();
   787→    // Exposed copper end
   788→    wire.fillStyle(0xcc8030, 0.9);
   789→    wire.fillCircle(x + 2, y, 2);
   790→    result.visualRects.push(wire as unknown as Phaser.GameObjects.Rectangle);
   791→
   792→    // Arcing spark burst (recurring — every 2-4 seconds)
   793→    const sparkBurst = () => {
   794→      if (!wire.active) return;
   795→      // Main flash
   796→      const flash = this.scene.add.circle(x + 2, y, 4 + Math.random() * 3, 0xc0e0ff, 0.9);
   797→      flash.setBlendMode(Phaser.BlendModes.ADD);
   798→      flash.setDepth(7);
   799→      this.trackedTween({
   800→        targets: flash, alpha: 0, scale: 2, duration: 100,
   801→        onComplete: () => flash.destroy(),
   802→      });
   803→      // Lightning bolt zig-zag (2-3 segments)
   804→      const bolt = this.scene.add.graphics();
   805→      bolt.setDepth(7);
   806→      bolt.lineStyle(1.5, 0xc0e0ff, 0.9);
   807→      bolt.beginPath();
   808→      bolt.moveTo(x + 2, y);
   809→      let bx = x + 2, by = y;
   810→      const segments = 2 + Math.floor(Math.random() * 2);
   811→      for (let s = 0; s < segments; s++) {
   812→        bx += (Math.random() - 0.5) * 12;
   813→        by += (Math.random() - 0.5) * 8;
   814→        bolt.lineTo(bx, by);
   815→      }
   816→      bolt.strokePath();
   817→      bolt.setBlendMode(Phaser.BlendModes.ADD);
   818→      this.trackedTween({
   819→        targets: bolt, alpha: 0, duration: 80,
   820→        onComplete: () => bolt.destroy(),
   821→      });
   822→      // Scattered spark particles
   823→      for (let i = 0; i < 4; i++) {
   824→        const px = x + 2 + (Math.random() - 0.5) * 8;
   825→        const py = y + (Math.random() - 0.5) * 6;
   826→        const p = this.scene.add.circle(px, py, 1, 0xc0e0ff, 1);
   827→        p.setBlendMode(Phaser.BlendModes.ADD).setDepth(7);
   828→        this.trackedTween({
   829→          targets: p,
   830→          x: px + (Math.random() - 0.5) * 20,
   831→          y: py + 4 + Math.random() * 8,
   832→          alpha: 0, duration: 300,
   833→          onComplete: () => p.destroy(),
   834→        });
   835→      }
   836→    };
   837→
   838→    // Schedule recurring sparks
   839→    const sparkTimer = this.scene.time.addEvent({
   840→      delay: 1500 + Math.random() * 2500,
   841→      loop: true,
   842→      callback: sparkBurst,
   843→    });
   844→    // Store the timer on the wire so it gets cleaned up when wire is destroyed
   845→    (wire as unknown as { __sparkTimer?: Phaser.Time.TimerEvent }).__sparkTimer = sparkTimer;
   846→    // Initial spark
   847→    this.scene.time.delayedCall(500 + Math.random() * 1000, sparkBurst);
   848→  }

  1312→  /** Destroy all loaded area objects. */
  1313→  unload(loaded: LoadedArea): void {
  1314→    // ── Stop all tracked tweens (sparks, fire, steam, lore pulses, etc.) ──
  1315→    this.activeTweens.forEach(t => { if (t && t.isPlaying()) t.stop(); });
  1316→    this.activeTweens = [];
  1317→    // Clean up hazard timers (electrical sparks, fire smoke, steam) before destroying visuals
  1318→    loaded.visualRects.forEach(v => {
  1319→      if (!v) return;
  1320→      const obj = v as unknown as { __sparkTimer?: Phaser.Time.TimerEvent; __smokeTimer?: Phaser.Time.TimerEvent; __steamTimer?: Phaser.Time.TimerEvent };
  1321→      obj.__sparkTimer?.remove();
  1322→      obj.__smokeTimer?.remove();
  1323→      obj.__steamTimer?.remove();
  1324→      if (v.active) v.destroy();
  1325→    });
  1326→    loaded.solids.forEach(s => { if (s && s.active) s.destroy(); });
  1327→    loaded.sectionTriggers.forEach(t => { if (t && t.active) t.destroy(); });
  1328→    loaded.checkpointTriggers.forEach(t => { if (t && t.active) t.destroy(); });
  1329→    loaded.hazardTriggers.forEach(t => { if (t && t.active) t.destroy(); });
  1330→    loaded.bossEntryTrigger?.destroy();
  1331→    loaded.loreObjects.forEach(l => { if (l && l.active) l.destroy(); });
  1332→    loaded.landmarks.forEach(l => { if (l && l.active) l.destroy(); });
  1333→    loaded.grappleAnchors.forEach(a => { if (a && a.active) a.destroy(); });
  1334→    // ── FIX Bug 1: Clean up physics bodies from EMP doors + shortcuts (not auto-destroyed) ──
  1335→    loaded.empDoors.forEach(d => {
  1336→      if (!d) return;
  1337→      const pb = d.getData('physicsBody') as Phaser.Physics.Matter.Image | null;
  1338→      if (pb && pb.active) {
  1339→        try { this.scene.matter.world.remove(pb.body as MatterJS.Body); } catch { /* */ }
  1340→        pb.destroy();
  1341→      }
  1342→      if (d.active) d.destroy();
  1343→    });
  1344→    loaded.shortcuts.forEach(s => {
  1345→      if (!s) return;
  1346→      const pb = s.getData('physicsBody') as Phaser.Physics.Matter.Image | null;
  1347→      if (pb && pb.active) {
  1348→        try { this.scene.matter.world.remove(pb.body as MatterJS.Body); } catch { /* */ }
  1349→        pb.destroy();
  1350→      }
  1351→      if (s.active) s.destroy();
  1352→    });
  1353→    loaded.collectibles.forEach(c => { if (c && c.active) c.destroy(); });
  1354→    loaded.solids = [];
  1355→    loaded.sectionTriggers = [];
  1356→    loaded.checkpointTriggers = [];
  1357→    loaded.hazardTriggers = [];
  1358→    loaded.bossEntryTrigger = null;
  1359→    loaded.visualRects = [];
  1360→    loaded.loreObjects = [];
  1361→    loaded.landmarks = [];
  1362→    loaded.grappleAnchors = [];
  1363→    loaded.empDoors = [];
  1364→    loaded.shortcuts = [];
  1365→    loaded.collectibles = [];
  1366→  }
