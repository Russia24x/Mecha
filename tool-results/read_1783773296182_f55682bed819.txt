   153‚Üí   153‚Üí  private setState(next: GameState): void {
   154‚Üí   154‚Üí    // Don't cleanup play when switching to overlay states (pause sub-menus)
   155‚Üí   155‚Üí    const isOverlay = ['settings', 'skills', 'inventory', 'quests', 'map'].includes(next);
   156‚Üí   156‚Üí    if (!isOverlay) {
   157‚Üí   157‚Üí      this.cleanupState();
   158‚Üí   158‚Üí    } else {
   159‚Üí   159‚Üí      // Just cleanup previous overlay container, keep play running
   160‚Üí   160‚Üí      if (this.stateContainer) { this.stateContainer.destroy(true); this.stateContainer = null; }
   161‚Üí   161‚Üí      if (this.menuNavHandler) { window.removeEventListener('keydown', this.menuNavHandler); this.menuNavHandler = null; }
   162‚Üí   162‚Üí    }
   163‚Üí   163‚Üí    this.state = next;
   164‚Üí   164‚Üí    this.menuButtons = [];
   165‚Üí   165‚Üí    this.menuFocusIndex = 0;
   166‚Üí   166‚Üí    if (!isOverlay && next !== 'play') {
   167‚Üí   167‚Üí      this.stateContainer = this.add.container(0, 0).setDepth(50);
   168‚Üí   168‚Üí    } else if (isOverlay) {
   169‚Üí   169‚Üí      this.stateContainer = this.add.container(0, 0).setDepth(250);
   170‚Üí   170‚Üí    } else {
   171‚Üí   171‚Üí      this.stateContainer = null;
   172‚Üí   172‚Üí    }
   173‚Üí   173‚Üí    switch (next) {
   174‚Üí   174‚Üí      case 'menu': this.buildMenu(); break;
   175‚Üí   175‚Üí      case 'hub': this.buildHub(); break;
   176‚Üí   176‚Üí      case 'play': this.buildPlay(); break;
   177‚Üí   177‚Üí      case 'map': this.buildMapOverlay(); break;
   178‚Üí   178‚Üí      case 'skills': this.buildSkillsOverlay(); break;
   179‚Üí   179‚Üí      case 'inventory': this.buildInventoryOverlay(); break;
   180‚Üí   180‚Üí      case 'quests': this.buildQuestsOverlay(); break;

   251‚Üí   251‚Üí  private buildMenu(): void {
   252‚Üí   252‚Üí    const c = this.stateContainer!;
   253‚Üí   253‚Üí    const w = GAME.WIDTH, h = GAME.HEIGHT;
   254‚Üí   254‚Üí
   255‚Üí   255‚Üí    // === Background: starry night sky ===
   256‚Üí   256‚Üí    const bg = this.add.graphics();
   257‚Üí   257‚Üí    bg.setDepth(0);
   258‚Üí   258‚Üí    // Deep blue gradient
   259‚Üí   259‚Üí    bg.fillStyle(0x040814, 1);
   260‚Üí   260‚Üí    bg.fillRect(0, 0, w, h);
   261‚Üí   261‚Üí    for (let r = 500; r > 0; r -= 30) {
   262‚Üí   262‚Üí      bg.fillStyle(0x0a1228, 0.02);
   263‚Üí   263‚Üí      bg.fillCircle(w / 2, h * 0.35, r);
   264‚Üí   264‚Üí    }
   265‚Üí   265‚Üí    c.add(bg);
   266‚Üí   266‚Üí
   267‚Üí   267‚Üí    // Stars ‚Äî 120 twinkling, drifting dots (more lively)
   268‚Üí   268‚Üí    for (let i = 0; i < 120; i++) {
   269‚Üí   269‚Üí      const sx = Math.random() * w;
   270‚Üí   270‚Üí      const sy = Math.random() * h * 0.75;
   271‚Üí   271‚Üí      const size = 0.5 + Math.random() * 2;
   272‚Üí   272‚Üí      const brightness = 0.2 + Math.random() * 0.8;
   273‚Üí   273‚Üí      const starColor = Math.random() < 0.15 ? 0xffe0a0 : Math.random() < 0.3 ? 0xa0c0ff : 0xc0e0ff;
   274‚Üí   274‚Üí      const star = this.add.circle(sx, sy, size, starColor, brightness);
   275‚Üí   275‚Üí      star.setDepth(1);
   276‚Üí   276‚Üí      c.add(star);
   277‚Üí   277‚Üí      // Twinkle ‚Äî varying intensity + duration
   278‚Üí   278‚Üí      this.tweens.add({
   279‚Üí   279‚Üí        targets: star,
   280‚Üí   280‚Üí        alpha: { from: brightness * 0.1, to: brightness },
   281‚Üí   281‚Üí        scale: { from: size * 0.5, to: size * 1.2 },
   282‚Üí   282‚Üí        duration: 600 + Math.random() * 2500,
   283‚Üí   283‚Üí        yoyo: true, repeat: -1,
   284‚Üí   284‚Üí        delay: Math.random() * 3000,
   285‚Üí   285‚Üí        ease: 'Sine.inOut',
   286‚Üí   286‚Üí      });
   287‚Üí   287‚Üí      // Slow drift ‚Äî each star moves slightly
   288‚Üí   288‚Üí      this.tweens.add({
   289‚Üí   289‚Üí        targets: star,
   290‚Üí   290‚Üí        x: sx + (Math.random() - 0.5) * 30,
   291‚Üí   291‚Üí        y: sy + (Math.random() - 0.5) * 20,
   292‚Üí   292‚Üí        duration: 5000 + Math.random() * 8000,

   693‚Üí   693‚Üí  private onCollisionStart = (event: MatterJS.Events.CollisionStartEvent): void => {
   694‚Üí   694‚Üí    for (const pair of event.pairs) {
   695‚Üí   695‚Üí      const aGo = (pair.bodyA as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   696‚Üí   696‚Üí      const bGo = (pair.bodyB as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   697‚Üí   697‚Üí      if (!aGo || !bGo) continue;
   698‚Üí   698‚Üí      const aIsPlayer = aGo.getData('entityType') === 'player';
   699‚Üí   699‚Üí      const bIsPlayer = bGo.getData('entityType') === 'player';
   700‚Üí   700‚Üí      // Section trigger
   701‚Üí   701‚Üí      const aSection = aGo.getData('sectionId') as number | undefined;
   702‚Üí   702‚Üí      const bSection = bGo.getData('sectionId') as number | undefined;
   703‚Üí   703‚Üí      if (aIsPlayer && bSection) { this.enterSection(bSection); }
   704‚Üí   704‚Üí      else if (bIsPlayer && aSection) { this.enterSection(aSection); }
   705‚Üí   705‚Üí      // Checkpoint
   706‚Üí   706‚Üí      else if (aIsPlayer && bGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   707‚Üí   707‚Üí      else if (bIsPlayer && aGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   708‚Üí   708‚Üí      // Boss entry
   709‚Üí   709‚Üí      else if (aIsPlayer && bGo.getData('isBossEntry')) { this.enterBossArena(); }
   710‚Üí   710‚Üí      else if (bIsPlayer && aGo.getData('isBossEntry')) { this.enterBossArena(); }
   711‚Üí   711‚Üí      // NPC interaction (handled via interact key, not collision)
   712‚Üí   712‚Üí      // Enemy contact
   713‚Üí   713‚Üí      if (aIsPlayer && bGo.getData('entityType') === 'enemy') { this.handleEnemyContact(bGo); }
   714‚Üí   714‚Üí      else if (bIsPlayer && aGo.getData('entityType') === 'enemy') { this.handleEnemyContact(aGo); }
   715‚Üí   715‚Üí      // Boss contact
   716‚Üí   716‚Üí      if (aIsPlayer && bGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   717‚Üí   717‚Üí      else if (bIsPlayer && aGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   718‚Üí   718‚Üí    }
   719‚Üí   719‚Üí  };

   780‚Üí   780‚Üí  private updatePlay(deltaMs: number): void {
   781‚Üí   781‚Üí    if (!this.player) return;
   782‚Üí   782‚Üí    this.player.update(deltaMs);
   783‚Üí   783‚Üí    this.render.update(this.time.now);
   784‚Üí   784‚Üí    this.hud?.update();
   785‚Üí   785‚Üí    // Projectiles
   786‚Üí   786‚Üí    for (let i = this.projectiles.length - 1; i >= 0; i--) {
   787‚Üí   787‚Üí      this.projectiles[i].update();
   788‚Üí   788‚Üí      if (!this.projectiles[i].isAlive) this.projectiles.splice(i, 1);
   789‚Üí   789‚Üí    }
   790‚Üí   790‚Üí    // Enemies
   791‚Üí   791‚Üí    const playerPos = this.player.position;
   792‚Üí   792‚Üí    for (let i = this.enemies.length - 1; i >= 0; i--) {
   793‚Üí   793‚Üí      const e = this.enemies[i];
   794‚Üí   794‚Üí      if (!e.isAlive || !e.sprite || !e.sprite.active) { this.enemies.splice(i, 1); continue; }
   795‚Üí   795‚Üí      try { e.update(deltaMs, playerPos); } catch { this.enemies.splice(i, 1); continue; }
   796‚Üí   796‚Üí    }
   797‚Üí   797‚Üí    // Boss
   798‚Üí   798‚Üí    if (this.boss && this.boss.isAlive && this.boss.sprite && this.boss.sprite.active) {
   799‚Üí   799‚Üí      try { this.boss.update(deltaMs); } catch { /* */ }
   800‚Üí   800‚Üí    }
   801‚Üí   801‚Üí    // Out of bounds
   802‚Üí   802‚Üí    if (this.player.sprite.y > GAME.HEIGHT + 80) {
   803‚Üí   803‚Üí      this.player.takeDamage(25);
   804‚Üí   804‚Üí      const area = WorldSystem.getCurrentArea();
   805‚Üí   805‚Üí      if (area) {
   806‚Üí   806‚Üí        const sec = area.sections.find(s => s.id === this.currentSection);
   807‚Üí   807‚Üí        if (sec) {
   808‚Üí   808‚Üí          this.player.sprite.setPosition(sec.x + 200, GAME.HEIGHT - 300);
   809‚Üí   809‚Üí          this.player.sprite.setVelocity(0, 0);
   810‚Üí   810‚Üí        }
   811‚Üí   811‚Üí      }
   812‚Üí   812‚Üí    }
   813‚Üí   813‚Üí    // Boss arena zoom
   814‚Üí   814‚Üí    if (this.bossArenaActive && this.boss && this.boss.isAlive) {
   815‚Üí   815‚Üí      this.camera.setZoom(0.85);
   816‚Üí   816‚Üí    }
   817‚Üí   817‚Üí  }

   819‚Üí   819‚Üí  private cleanupPlay(): void {
   820‚Üí   820‚Üí    this.matter.world.off('collisionstart', this.onCollisionStart, this);
   821‚Üí   821‚Üí    this.projectiles.forEach(p => p.kill());
   822‚Üí   822‚Üí    this.projectiles = [];
   823‚Üí   823‚Üí    this.player?.destroy();
   824‚Üí   824‚Üí    this.enemies.forEach(e => e.destroy());
   825‚Üí   825‚Üí    this.enemies = [];
   826‚Üí   826‚Üí    this.boss?.destroy();
   827‚Üí   827‚Üí    this.boss = null;
   828‚Üí   828‚Üí    if (this.loadedArea && this.areaLoader) this.areaLoader.unload(this.loadedArea);
   829‚Üí   829‚Üí    this.loadedArea = null;
   830‚Üí   830‚Üí    this.tweens.killAll();
   831‚Üí   831‚Üí    this.sequenceTimers.forEach(t => t.remove());
   832‚Üí   832‚Üí    this.sequenceTimers = [];
   833‚Üí   833‚Üí    this.hud?.destroy();
   834‚Üí   834‚Üí    this.render?.destroy();
   835‚Üí   835‚Üí    this.camera.resetZoom();
   836‚Üí   836‚Üí    this.camera.stopFollow();
   837‚Üí   837‚Üí    this.camera.setBounds(0, 0, GAME.WIDTH, GAME.HEIGHT);
   838‚Üí   838‚Üí    this.physics.setWorldBounds(GAME.WIDTH, GAME.HEIGHT);
   839‚Üí   839‚Üí    // Note: InputSystem.destroy() only called in scene shutdown, not cleanupPlay.
   840‚Üí   840‚Üí    // PlayerEntity.destroy() handles its own input cleanup.
   841‚Üí   841‚Üí    this.paused = false;
   842‚Üí   842‚Üí  }

   846‚Üí   846‚Üí  private togglePause(): void {

   848‚Üí   848‚Üí    const now = this.time.now;
   849‚Üí   849‚Üí    if (now - this.lastPauseToggleAt < 300) return;
   850‚Üí   850‚Üí    this.lastPauseToggleAt = now;
   851‚Üí   851‚Üí
   852‚Üí   852‚Üí    if (this.paused) {
