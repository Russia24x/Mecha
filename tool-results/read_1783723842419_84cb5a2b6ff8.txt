    82→  setState(next: GameState): void {
    83→    this.cleanupState();
    84→    this.state = next;
    85→    // Only create stateContainer for menu states (not play)
    86→    if (next !== 'play') {
    87→      this.stateContainer = this.add.container(0, 0).setDepth(50);
    88→    } else {
    89→      this.stateContainer = null;
    90→    }
    91→    this.menuButtons = [];
    92→    this.menuFocusIndex = 0;
    93→    switch (next) {
    94→      case 'menu': this.buildMenu(); break;
    95→      case 'map': this.buildMap(); break;
    96→      case 'play': this.buildPlay(); break;
    97→      case 'victory': this.buildVictory(); break;
    98→      case 'comingsoon': this.buildComingSoon(); break;
    99→      case 'skills': this.buildSkills(); break;
   100→      case 'settings': this.buildSettings(); break;
   101→    }
   102→    // Focus first button for keyboard/gamepad nav
   103→    if (this.menuButtons.length > 0) this.focusButton(0);
   104→  }
   105→
   106→  private cleanupState(): void {
   107→    if (this.state === 'play') this.cleanupPlay();
   108→    // Remove menu keyboard handler when leaving any state
   109→    if (this.menuNavHandler) {
   110→      window.removeEventListener('keydown', this.menuNavHandler);
   111→      this.menuNavHandler = null;
   112→    }
   113→    if (this.stateContainer) { this.stateContainer.destroy(true); this.stateContainer = null; }
   114→  }

   116→  update(_time: number, deltaMs: number): void {
   117→    if (this.state === 'play') {
   118→      this.updatePlay(deltaMs);
   119→    } else {
   120→      // Menu navigation for non-play states
   121→      GamepadManager.update();
   122→      this.updateMenuNavigation(deltaMs);
   123→    }
   124→  }

   254→  private buildPlay(): void {
   255→    const sd = this.stageData;

   260→    this.matter.world.setGravity(0, 0.9);
   261→    this.projectiles = []; this.enemies = []; this.boss = null;
   262→    this.solids = []; this.sectionTriggers = []; this.bossArenaActive = false;
   263→    this.healthPickups = [];
   264→    this.stageStartTime = this.time.now;
   265→
   266→    // Stage 2 background — texture already preloaded in BootScene
   267→    if (this.currentStageId === 2 && this.textures.exists('stage2-bg')) {
   268→      const bg = this.add.image(GAME.WIDTH / 2, GAME.HEIGHT / 2, 'stage2-bg')
   269→        .setDepth(-2)
   270→        .setDisplaySize(GAME.WIDTH, GAME.HEIGHT)
   271→        .setAlpha(0.5)
   272→        .setScrollFactor(0); // fixed — doesn't scroll with camera
   273→      this.stageBgImage = bg;
   274→      // Dark overlay for gameplay contrast
   275→      const dark = this.add.rectangle(0, 0, GAME.WIDTH, GAME.HEIGHT, sd.bgColor, 0.3)
   276→        .setOrigin(0, 0).setDepth(-1.5).setScrollFactor(0);
   277→    }
   278→
   279→    // Only create parallax for stage 1 (stage 2 uses bg image instead)
   280→    if (this.currentStageId === 1) {
   281→      this.parallax = new Parallax(this);
   282→    }
   283→    this.graphics = new Graphics(this);
   284→    this.buildStageGeometry();
   285→    this.buildTutorialHints();
   286→    this.damageSystem = new DamageSystem(this);
   287→    const startX = (this.currentSection - 1) * sd.SECTION_WIDTH + 200;
   288→    this.player = new Player(this, startX, GAME.HEIGHT - 300, this.projectiles, this.damageSystem);
   289→    this.cameras.main.startFollow(this.player.sprite, true, 0.1, 0.1);
   290→    this.cameras.main.setDeadzone(160, 100);
   291→    this.graphics.addLight({
   292→      follow: () => {
   293→        if (!this.player || !this.player.sprite || !this.player.sprite.active) return new Phaser.Math.Vector2(0, 0);
   294→        return this.player.position;
   295→      },
   296→      radius: 180, color: COLORS.PLAYER_GLOW, intensity: 0.4, flicker: 0.05,
   297→    });
   298→    this.hud = new HUD(this, this.player);
   299→    this.bossBar = new BossBar(this, this.boss);
   300→    FloatingText.init(this);
   301→    this.spawnEnemiesForSection(this.currentSection);
   302→    this.buildSectionTriggers();
   303→    EventBus.on('PLAYER_DEAD', this.onPlayerDied, this);
   304→    EventBus.on('ENEMY_DEAD', this.onEnemyKilled, this);
   305→    EventBus.on('BOSS_PHASE', this.onBossPhaseEvent, this);
   306→    EventBus.on('PLAYER_DAMAGED', this.onPlayerDamaged, this);
   307→
   308→    const sec = sd.SECTIONS[this.currentSection - 1];
   309→    EventBus.emit('GAME_STATE', { sectionId: sec.id, sectionName: sec.name });
   310→  }

   312→  private updatePlay(deltaMs: number): void {
   313→    if (!this.player) return;
   314→    this.player.update(deltaMs);
   315→    this.graphics.update(this.time.now, deltaMs);
   316→    this.hud?.update();
   317→    this.bossBar?.update();
   318→    // Projectiles
   319→    for (let i = this.projectiles.length - 1; i >= 0; i--) {
   320→      const p = this.projectiles[i];
   321→      p.update(this);
   322→      if (!p.isAlive) this.projectiles.splice(i, 1);
   323→    }
   324→    // Enemies
   325→    const playerPos = this.player.position;
   326→    const cx = this.cameras.main.scrollX + GAME.WIDTH / 2;
   327→    for (let i = this.enemies.length - 1; i >= 0; i--) {
   328→      const e = this.enemies[i];
   329→      if (!e.isAlive || !e.sprite || !e.sprite.active) { this.enemies.splice(i, 1); continue; }
   330→      if (Math.abs(e.sprite.x - cx) > GAME.WIDTH * 1.2) continue;
   331→      try { e.update(deltaMs, playerPos); } catch { this.enemies.splice(i, 1); continue; }
   332→      if (!e.isAlive) this.enemies.splice(i, 1);
   333→    }
   334→    // Boss
   335→    if (this.boss && this.boss.isAlive && this.boss.sprite && this.boss.sprite.active) {
   336→      try { this.boss.update(deltaMs); } catch { /* */ }
   337→    }
   338→    this.parallax?.update(deltaMs);
   339→    Ragdoll.update(this);
   340→    // Out-of-bounds
   341→    if (this.player.sprite.y > GAME.HEIGHT + 80) {
   342→      this.player.takeDamage(25);
   343→      this.respawnPlayerAtSection();
   344→    }
   345→    // Boss arena zoom
   346→    if (this.bossArenaActive && this.boss && this.boss.isAlive) {
   347→      this.cameras.main.setZoom(Phaser.Math.Linear(this.cameras.main.zoom, 0.85, 0.04));
   348→    }
   349→  }

   351→  private cleanupPlay(): void {
   352→    EventBus.off('PLAYER_DEAD', this.onPlayerDied, this);
   353→    EventBus.off('ENEMY_DEAD', this.onEnemyKilled, this);
   354→    EventBus.off('BOSS_PHASE', this.onBossPhaseEvent, this);
   355→    EventBus.off('PLAYER_DAMAGED', this.onPlayerDamaged, this);
   356→
   357→    this.matter.world.off('collisionstart', this.onCollisionStart);
   358→    this.projectiles.forEach(p => p.kill());
   359→    this.projectiles = [];
   360→    this.hud?.destroy();
   361→    this.bossBar?.destroy();
   362→    this.parallax?.destroy();
   363→    this.graphics?.destroy();
   364→    Ragdoll.clear(this);
   365→    this.tutorialHints.forEach(h => h.destroy());
   366→    this.tutorialHints = [];
   367→    this.healthPickups.forEach(h => h.destroy());
   368→    this.healthPickups = [];
   369→    this.stageBgImage?.destroy();
   370→    this.stageBgImage = null;
   371→    FloatingText.destroy();
   372→    this.cameras.main.setZoom(1);
   373→    this.cameras.main.stopFollow();
   374→  }

   414→        this.addDestructibleCrate(baseX + 600, GAME.HEIGHT - 60);
   415→        this.addDestructibleCrate(baseX + 750, GAME.HEIGHT - 60);
   416→        break;

   499→  private onCollisionStart = (event: MatterJS.Events.CollisionStartEvent) => {
   500→    for (const pair of event.pairs) {
   501→      const aGo = (pair.bodyA as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   502→      const bGo = (pair.bodyB as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   503→      if (!aGo || !bGo) continue;
   504→      const aIsPlayer = aGo.getData('entityType') === 'player';
   505→      const bIsPlayer = bGo.getData('entityType') === 'player';
   506→      const aSection = aGo.getData('sectionId') as number | undefined;
   507→      const bSection = bGo.getData('sectionId') as number | undefined;
   508→      if (aIsPlayer && bSection) this.enterSection(bSection);
   509→      else if (bIsPlayer && aSection) this.enterSection(aSection);
   510→      else if (aIsPlayer && bGo.getData('isCheckpoint')) this.activateCheckpoint();
   511→      else if (bIsPlayer && aGo.getData('isCheckpoint')) this.activateCheckpoint();
   512→      else if (aIsPlayer && bGo.getData('isBossEntry')) this.enterBossArena();
   513→      else if (bIsPlayer && aGo.getData('isBossEntry')) this.enterBossArena();
   514→      // Health pickup
   515→      else if (aIsPlayer && bGo.getData('isHealthPickup')) this.collectHealthPickup(bGo);
   516→      else if (bIsPlayer && aGo.getData('isHealthPickup')) this.collectHealthPickup(aGo);
   517→      // Spike trap
   518→      else if (aIsPlayer && bGo.getData('isSpikeTrap')) this.player.takeDamage(15);
   519→      else if (bIsPlayer && aGo.getData('isSpikeTrap')) this.player.takeDamage(15);
   520→      if (aIsPlayer && bGo.getData('entityType') === 'enemy') this.handleEnemyContact(bGo);
   521→      else if (bIsPlayer && aGo.getData('entityType') === 'enemy') this.handleEnemyContact(aGo);
   522→      if (aIsPlayer && bGo.getData('entityType') === 'boss' && this.boss) this.player.takeDamage(this.boss.getContactDamage());
   523→      else if (bIsPlayer && aGo.getData('entityType') === 'boss' && this.boss) this.player.takeDamage(this.boss.getContactDamage());
   524→    }
   525→  };

   768→  private onEnemyKilled = (payload: { id: string; score: number }): void => {
   769→    Save.recordKill();
   770→    // XP reward based on enemy type
   771→    let xpReward = 15;
   772→    if (payload.id.startsWith('drone-')) xpReward = 15;
   773→    else if (payload.id.startsWith('spider-')) xpReward = 25;
   774→    else if (payload.id.startsWith('heavy-')) xpReward = 50;
   775→    const xpResult = SkillTree.recordKill(xpReward);
   776→    Effects.play('enemyHit');
   777→    FloatingText.onKill();
   778→    if (xpResult.leveledUp) {

   780→      this.hud.toast(`LEVEL UP! Lv.${data.level} — +1 SP`);
   781→      Effects.screenFlash(this, 0xffe060, 0.3, 400);
   782→    }
   783→    const enemy = this.enemies.find(e => e.id === payload.id);
   784→    let enemyX = 0, enemyY = 0;
   785→    if (enemy) {
   786→      enemyX = enemy.sprite.x;
   787→      enemyY = enemy.sprite.y;
   788→      Effects.explosion(this, enemyX, enemyY, COLORS.ENEMY_DRONE, 0.6);
   789→    }
   790→    // Progressive weapon unlock based on enemy type
   791→    if (payload.id.startsWith('spider-') && !this.player.hasWeapon('shotgun')) {
   792→      this.player.unlockWeapon('shotgun');
   793→      this.hud.toast('SHOTGUN UNLOCKED — press 2');
   794→    } else if (payload.id.startsWith('heavy-') && !this.player.hasWeapon('rocket')) {
   795→      this.player.unlockWeapon('rocket');
   796→      this.hud.toast('ROCKET UNLOCKED — press 4');
   797→    } else if (payload.id.startsWith('drone-') && !this.player.hasWeapon('laser')) {
   798→      // Laser unlocks when all drones in the section are cleared
   799→      const dronesLeft = this.enemies.filter(e => e.id.startsWith('drone-') && e.isAlive).length;
   800→      if (dronesLeft === 0) {
   801→        this.player.unlockWeapon('laser');
   802→        this.hud.toast('LASER UNLOCKED — press 3');
   803→      }
   804→    }
   805→    // Health pickup — 35% chance to drop
   806→    if (enemyX > 0 && Math.random() < 0.35) {
   807→      this.spawnHealthPickup(enemyX, enemyY);
   808→    }
   809→  };

  1143→      const v = (clamped - x - 20) / 240;
  1144→      fill.width = 240 * v;
  1145→      if (label === 'Master Volume') Effects.setMasterVolume(v);
  1146→      else if (label === 'Music Volume') Effects.setMusicVolume(v);
  1147→      else if (label === 'SFX Volume') Effects.setSfxVolume(v);

  1150→  }

  1194→  /** Navigate menu with keyboard arrows + Enter, or gamepad D-pad/stick + A. */
  1195→  private menuNavHandler: ((e: KeyboardEvent) => void) | null = null;
  1196→
  1197→  private updateMenuNavigation(deltaMs: number): void {
  1198→    if (this.menuButtons.length === 0) return;
  1199→    this.menuNavCooldown -= deltaMs;
  1200→
  1201→    // Ensure a button is always focused
  1202→    if (this.menuFocusIndex >= this.menuButtons.length) this.menuFocusIndex = 0;
  1203→    if (this.menuFocusIndex < 0) this.menuFocusIndex = 0;
  1204→
  1205→    // Set up window keyboard listener once per state
  1206→    if (!this.menuNavHandler) {
  1207→      this.menuNavHandler = (e: KeyboardEvent) => {
  1208→        if (this.state === 'play') return; // skip during play
  1209→        if (this.menuButtons.length === 0) return;
  1210→        switch (e.code) {
  1211→          case 'KeyW': case 'ArrowUp':
  1212→            this.focusButton((this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length);
  1213→            Effects.play('uiHover');
  1214→            break;
  1215→          case 'KeyS': case 'ArrowDown':
  1216→            this.focusButton((this.menuFocusIndex + 1) % this.menuButtons.length);
  1217→            Effects.play('uiHover');
  1218→            break;
  1219→          case 'Enter': case 'Space': {
  1220→            const btn = this.menuButtons[this.menuFocusIndex];
  1221→            if (btn) { Effects.play('uiClick'); btn.onClick(); }
  1222→            break;
  1223→          }
  1224→          case 'Escape':
  1225→            this.menuGoBack();
  1226→            break;
  1227→        }
  1228→      };
  1229→      window.addEventListener('keydown', this.menuNavHandler);
  1230→    }
  1231→
  1232→    // Gamepad: Left stick OR D-pad to navigate, A to activate, B to go back
  1233→    const gp = GamepadManager.getState();
  1234→    if (this.menuNavCooldown <= 0) {
  1235→      if (gp.leftStickY < -0.3) {
  1236→        this.focusButton((this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length);
  1237→        Effects.play('uiHover');
  1238→        this.menuNavCooldown = 200;
  1239→      } else if (gp.leftStickY > 0.3) {
  1240→        this.focusButton((this.menuFocusIndex + 1) % this.menuButtons.length);
  1241→        Effects.play('uiHover');
  1242→        this.menuNavCooldown = 200;
  1243→      }
  1244→    }
  1245→    if (gp.jumpPressed) {
  1246→      const btn = this.menuButtons[this.menuFocusIndex];
  1247→      if (btn) { Effects.play('uiClick'); btn.onClick(); }
  1248→    }
  1249→    if (gp.backPressed) {
  1250→      this.menuGoBack();
  1251→    }
  1252→  }
  1253→
  1254→  /** Go back from current menu state. */
  1255→  private menuGoBack(): void {
  1256→    Effects.play('uiClick');
  1257→    switch (this.state) {
  1258→      case 'map': this.setState('menu'); break;
  1259→      case 'skills': this.setState('map'); break;
  1260→      case 'settings': this.setState('map'); break;
  1261→      case 'victory': this.setState('map'); break;
  1262→      case 'comingsoon': this.setState('map'); break;
  1263→      // menu: no back (it's the root)
  1264→    }
  1265→  }
