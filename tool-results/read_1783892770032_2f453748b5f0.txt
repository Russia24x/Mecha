   146‚Üí   146‚Üí  private computeStats(unlockedSkills: string[]): PlayerStats {
   147‚Üí   147‚Üí    const base: PlayerStats = {
   148‚Üí   148‚Üí      maxHealth: PLAYER.MAX_HEALTH,
   149‚Üí   149‚Üí      maxEnergy: PLAYER.MAX_ENERGY,
   150‚Üí   150‚Üí      energyRegen: PLAYER.ENERGY_REGEN,
   151‚Üí   151‚Üí      moveSpeed: PLAYER.MOVE_SPEED,
   152‚Üí   152‚Üí      jumpVelocity: PLAYER.JUMP_VELOCITY,
   153‚Üí   153‚Üí      dashSpeed: PLAYER.DASH_SPEED,
   154‚Üí   154‚Üí      dashDurationMs: PLAYER.DASH_DURATION_MS,
   155‚Üí   155‚Üí      dashCooldownMs: PLAYER.DASH_COOLDOWN_MS,
   156‚Üí   156‚Üí      meleeDamage: PLAYER.MELEE_DAMAGE,
   157‚Üí   157‚Üí      meleeRange: PLAYER.MELEE_RANGE,
   158‚Üí   158‚Üí      fireCooldownMs: PLAYER.FIRE_COOLDOWN_MS,
   159‚Üí   159‚Üí      invulnMs: PLAYER.INVULN_MS,
   160‚Üí   160‚Üí    };
   161‚Üí   161‚Üí
   162‚Üí   162‚Üí    // Apply each unlocked skill's effect
   163‚Üí   163‚Üí    for (const skillId of unlockedSkills) {
   164‚Üí   164‚Üí      const skill = getSkill(skillId);
   165‚Üí   165‚Üí      if (!skill) continue;
   166‚Üí   166‚Üí      const eff = skill.effect;
   167‚Üí   167‚Üí      const statKey = eff.stat as keyof PlayerStats;
   168‚Üí   168‚Üí      if (eff.multiplier && typeof base[statKey] === 'number') {
   169‚Üí   169‚Üí        (base[statKey] as number) *= eff.multiplier;
   170‚Üí   170‚Üí      }
   171‚Üí   171‚Üí      if (eff.additive && typeof base[statKey] === 'number') {
   172‚Üí   172‚Üí        (base[statKey] as number) += eff.additive;
   173‚Üí   173‚Üí      }
   174‚Üí   174‚Üí    }
   175‚Üí   175‚Üí
   176‚Üí   176‚Üí    return base;
   177‚Üí   177‚Üí  }

   180‚Üí   180‚Üí  public refreshStats(): void {
   181‚Üí   181‚Üí    const save = SaveSystem.getPlayer();
   182‚Üí   182‚Üí    const oldMaxHp = this.stats.maxHealth;
   183‚Üí   183‚Üí    const oldMaxEn = this.stats.maxEnergy;
   184‚Üí   184‚Üí    this.stats = this.computeStats(save.unlockedSkills);
   185‚Üí   185‚Üí    this.abilities = new Set(save.abilities);
   186‚Üí   186‚Üí    this.maxJumps = this.abilities.has('doubleJump') ? 2 : 1;
   187‚Üí   187‚Üí    // Adjust current health/energy to new max
   188‚Üí   188‚Üí    const hpDiff = this.stats.maxHealth - oldMaxHp;
   189‚Üí   189‚Üí    const enDiff = this.stats.maxEnergy - oldMaxEn;
   190‚Üí   190‚Üí    this.health.max = this.stats.maxHealth;
   191‚Üí   191‚Üí    this.energy.max = this.stats.maxEnergy;
   192‚Üí   192‚Üí    this.energy.regenPerSec = this.stats.energyRegen;
   193‚Üí   193‚Üí    if (hpDiff > 0) this.health.current += hpDiff;
   194‚Üí   194‚Üí    if (enDiff > 0) this.energy.current += enDiff;
   195‚Üí   195‚Üí    this.health.current = Math.min(this.health.current, this.health.max);
   196‚Üí   196‚Üí    this.energy.current = Math.min(this.energy.current, this.energy.max);
   197‚Üí   197‚Üí  }

   466‚Üí   466‚Üí  private tryFire(): void {
   467‚Üí   467‚Üí    if (!this.alive || !this.sprite || !this.sprite.active) return;
   468‚Üí   468‚Üí    const now = this.scene.time.now;
   469‚Üí   469‚Üí    const weapon = getWeapon(this.currentWeapon);

   471‚Üí   471‚Üí    if (now - this.lastFireAt < cooldown) return;

   474‚Üí   474‚Üí    this.lastFireAt = now;
   475‚Üí   475‚Üí    const from = this.position;
   476‚Üí   476‚Üí    const aimDir = this.getAimDirection();
   477‚Üí   477‚Üí    const muzzle = new Phaser.Math.Vector2(from.x + aimDir.x * 30, from.y - 6 + aimDir.y * 30);

   479‚Üí   479‚Üí
   480‚Üí   480‚Üí    // Hitscan weapons (railgun, laser)

   482‚Üí   482‚Üí      this.fireHitscan(weapon, muzzle, aimDir);
   483‚Üí   483‚Üí      return;
   484‚Üí   484‚Üí    }
   485‚Üí   485‚Üí
   486‚Üí   486‚Üí    // Projectile weapons
   487‚Üí   487‚Üí    const bulletCount = weapon.bulletsPerShot ?? 1;

   489‚Üí   489‚Üí      let angleOffset = 0;
   490‚Üí   490‚Üí      if (bulletCount > 1 && weapon.spread) {
   491‚Üí   491‚Üí        const t = i / (bulletCount - 1) - 0.5;
   492‚Üí   492‚Üí        angleOffset = t * weapon.spread;
   493‚Üí   493‚Üí      }
   494‚Üí   494‚Üí      const angle = Math.atan2(aimDir.y, aimDir.x) + angleOffset;
   495‚Üí   495‚Üí      const dir = new Phaser.Math.Vector2(Math.cos(angle), Math.sin(angle));
   496‚Üí   496‚Üí      const proj = new Projectile(this.scene, this.physics, this.particles, muzzle, dir, {
   497‚Üí   497‚Üí        speed: weapon.bulletSpeed ?? PLAYER.BULLET_SPEED,
   498‚Üí   498‚Üí        damage: this.currentWeapon === 'assault_rifle' ? PLAYER.BULLET_DAMAGE : weapon.damage,
   499‚Üí   499‚Üí        ttl: 1500, owner: 'player', color: weapon.color, size: weapon.size,
   500‚Üí   500‚Üí        weapon, explosive: weapon.tier === 'explosive', explosionRadius: weapon.explosionRadius,
   501‚Üí   501‚Üí      });
   502‚Üí   502‚Üí      this.projectiles.push(proj);
   503‚Üí   503‚Üí    }
   504‚Üí   504‚Üí    this.particles.sparks(muzzle.x, muzzle.y, weapon.color, (weapon.bulletsPerShot ?? 1) + 2);
   505‚Üí   505‚Üí  }
