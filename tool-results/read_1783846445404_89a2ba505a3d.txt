   166→  // ================ STATE MACHINE ================
   167→
   168→  private setState(next: GameState): void {
   169→    // Cleanup previous state
   170→    this.cleanupState();
   171→    this.state = next;
   172→    this.menuButtons = [];
   173→    this.menuFocusIndex = 0;
   174→    this.stateContainer = this.add.container(0, 0).setDepth(50);
   175→    switch (next) {
   176→      case 'menu': this.buildMenu(); break;
   177→      case 'hub': this.buildHub(); break;
   178→      case 'play': this.buildPlay(); break;
   179→      case 'gameover': this.buildGameOver(); break;
   180→      case 'victory': this.buildVictory(); break;
   181→    }
   182→    // setScrollFactor(0,0,true) AFTER all children are added by build* methods
   183→    this.stateContainer.setScrollFactor(0, 0, true);
   184→  }
   185→
   186→  private cleanupState(): void {
   187→    // Close any open overlays first
   188→    OverlayManager.closeAll();
   189→    // Destroy HUD if leaving play (hub is a separate environment — no HUD)
   190→    if (this.state === 'play') {
   191→      this.cleanupPlay();
   192→    }
   193→    if (this.menuNavHandler) {
   194→      window.removeEventListener('keydown', this.menuNavHandler);
   195→      this.menuNavHandler = null;
   196→    }
   197→    if (this.stateContainer) {
   198→      this.stateContainer.destroy(true);
   199→      this.stateContainer = null;
   200→    }
   201→    // Hide pause menu if visible
   202→    if (this.pauseMenuUI?.isVisible) this.pauseMenuUI.hide();
   203→    this.paused = false;
   204→  }

   206→  // ================ OVERLAY MANAGEMENT ================
   207→
   208→  /**
   209→   * Open an overlay from the current context.
   210→   * - From hub: overlay sits on top of hub. Closing returns to hub.
   211→   * - From play (paused): hide pause menu first. Closing reopens pause menu.
   212→   */
   213→  private openOverlay(id: OverlayId): void {
   214→    const parent: OverlayParent = this.state === 'hub' ? 'hub' : this.state === 'play' ? 'play' : 'menu';
   215→
   216→    // If from play, hide pause menu (paused stays true so play doesn't update)
   217→    if (parent === 'play') {
   218→      this.pauseMenuUI.hide();
   219→    }
   220→
   221→    // Build the overlay UI on demand
   222→    let ui: OverlayUI | null = null;
   223→    switch (id) {
   224→      case 'settings':
   225→        ui = new SettingsUI(this, () => this.closeOverlay());
   226→        break;
   227→      case 'skills':
   228→        ui = new SkillTreeUI(this, () => this.closeOverlay());
   229→        break;
   230→      case 'inventory':
   231→        ui = new InventoryUI(this, () => this.closeOverlay());
   232→        break;
   233→      case 'quests':
   234→        ui = new QuestUI(this, () => this.closeOverlay());
   235→        break;
   236→      case 'map':
   237→        ui = new WorldMapUI(this,
   238→          () => this.closeOverlay(),
   239→          (areaId: string) => this.fastTravel(areaId),
   240→        );
   241→        break;
   242→    }
   243→    if (!ui) return;
   244→    OverlayManager.open(id, ui, parent);
   245→  }
   246→
   247→  /** Close the current overlay and return to its parent. */
   248→  private closeOverlay(): void {
   249→    OverlayManager.close((parent) => {
   250→      if (parent === 'play') {
   251→        // Reopen pause menu
   252→        this.pauseMenuUI.show();
   253→      }
   254→      // If parent === 'hub', hub is still visible underneath — nothing to do
   255→    });
   256→  }

   258→  // ================ UPDATE LOOP ================
   259→
   260→  update(_time: number, deltaMs: number): void {
   261→    InputSystem.update();
   262→    const input = InputSystem.getState();
   263→
   264→    // *** Overlay input has highest priority — B/ESC closes, gamepad navigates ***
   265→    if (OverlayManager.hasOpen) {
   266→      OverlayManager.handleInput((parent) => {
   267→        if (parent === 'play') {
   268→          this.pauseMenuUI.show();
   269→        }
   270→      });
   271→      this.handleDialogueInput(input);
   272→      return;  // Block all other input while overlay is open
   273→    }
   274→
   275→    if (this.state === 'play') {
   276→      // ESC / Start = toggle pause
   277→      if (input.pausePressed) {
   278→        this.togglePause();
   279→      }
   280→      if (!this.paused) {
   281→        if (input.interactPressed) this.tryInteract();
   282→        this.updatePlay(deltaMs);
   283→      } else {
   284→        // Paused — handle pause menu navigation
   285→        this.pauseMenuUI.handleNavigation();
   286→      }
   287→    } else if (this.state === 'menu' || this.state === 'hub' || this.state === 'gameover' || this.state === 'victory') {
   288→      // Gamepad + keyboard navigation
   289→      this.handleMenuGamepadNav(input);
   290→      // ESC in hub = back to menu
   291→      if (this.state === 'hub' && input.pausePressed) {
   292→        this.setState('menu');
   293→      }
   294→    }
   295→
   296→    this.handleDialogueInput(input);
   297→  }
   298→
   299→  private handleDialogueInput(input: import('../../systems/InputSystem').InputState): void {
   300→    if (this.dialogueUI?.isVisible) {
   301→      if (input.jumpPressed || input.firePressed) {
   302→        this.dialogueUI.advance();
   303→      }
   304→    }
   305→  }

   309→  private buildMenu(): void {

   311→    const w = GAME.WIDTH, h = GAME.HEIGHT;

   316→    bg.fillStyle(0x040814, 1);
   317→    bg.fillRect(0, 0, w, h);
   318→    for (let r = 500; r > 0; r -= 30) {
   319→      bg.fillStyle(0x0a1228, 0.02);
   320→      bg.fillCircle(w / 2, h * 0.35, r);
   321→    }
   322→    c.add(bg);
   323→
   324→    // Stars — 120 twinkling, drifting dots
   325→    for (let i = 0; i < 120; i++) {
   326→      const sx = Math.random() * w;
   327→      const sy = Math.random() * h * 0.75;
   328→      const size = 0.5 + Math.random() * 2;
   329→      const brightness = 0.2 + Math.random() * 0.8;
   330→      const starColor = Math.random() < 0.15 ? 0xffe0a0 : Math.random() < 0.3 ? 0xa0c0ff : 0xc0e0ff;
   331→      const star = this.add.circle(sx, sy, size, starColor, brightness);
   332→      star.setDepth(1);
   333→      c.add(star);
   334→      this.tweens.add({
   335→        targets: star,
   336→        alpha: { from: brightness * 0.1, to: brightness },
   337→        scale: { from: size * 0.5, to: size * 1.2 },
   338→        duration: 600 + Math.random() * 2500,
   339→        yoyo: true, repeat: -1,
   340→        delay: Math.random() * 3000,
   341→        ease: 'Sine.inOut',
   342→      });
   343→      this.tweens.add({
   344→        targets: star,
   345→        x: sx + (Math.random() - 0.5) * 30,
   346→        y: sy + (Math.random() - 0.5) * 20,
   347→        duration: 5000 + Math.random() * 8000,
   348→        yoyo: true, repeat: -1,
   349→        ease: 'Sine.inOut',
   350→      });

   353→    // Shooting stars

   366→        targets: ss,
   367→        x: endX, y: endY,
   368→        alpha: { from: 0.9, to: 0 },
   369→        duration: 600 + Math.random() * 400,
   370→        onComplete: () => ss.destroy(),
   371→      });
   372→    };
   373→    this.time.addEvent({
   374→      delay: 3000, loop: true,
   375→      callback: () => { if (Math.random() < 0.4) shootingStarFunc(); },

   377→
   378→    // Brighter "beacon" stars with pulsing glow

   380→      const bx = Math.random() * w;
   381→      const by = Math.random() * h * 0.55;

   390→        targets: beaconGlow,
   391→        alpha: { from: 0.05, to: 0.25 },

   397→        targets: beacon,
   398→        alpha: { from: 0.6, to: 1 },

   439→    // === Footer ===

   665→  private onCollisionStart = (event: MatterJS.IEventCollision<MatterJS.Body>): void => {
   666→    for (const pair of event.pairs) {
   667→      const aGo = (pair.bodyA as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   668→      const bGo = (pair.bodyB as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   669→      if (!aGo || !bGo) continue;
   670→      const aIsPlayer = aGo.getData('entityType') === 'player';
   671→      const bIsPlayer = bGo.getData('entityType') === 'player';
   672→      const aSection = aGo.getData('sectionId') as number | undefined;
   673→      const bSection = bGo.getData('sectionId') as number | undefined;
   674→      if (aIsPlayer && bSection) { this.enterSection(bSection); }
   675→      else if (bIsPlayer && aSection) { this.enterSection(aSection); }
   676→      else if (aIsPlayer && bGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   677→      else if (bIsPlayer && aGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   678→      else if (aIsPlayer && bGo.getData('isBossEntry')) { this.enterBossArena(); }
   679→      else if (bIsPlayer && aGo.getData('isBossEntry')) { this.enterBossArena(); }
   680→      if (aIsPlayer && bGo.getData('entityType') === 'enemy') { this.handleEnemyContact(bGo); }
   681→      else if (bIsPlayer && aGo.getData('entityType') === 'enemy') { this.handleEnemyContact(aGo); }
   682→      if (aIsPlayer && bGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   683→      else if (bIsPlayer && aGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   684→    }
   685→  };

   716→  private handleEnemyContact(enemyGo: Phaser.GameObjects.GameObject): void {
   717→    const id = enemyGo.getData('id') as string | undefined;
   718→    const dmg = id?.startsWith('drone-') ? 8 : id?.startsWith('spider-') ? 14 : id?.startsWith('heavy-') ? 22 : id?.startsWith('sniper-') ? 12 : 10;
   719→    if (this.player.takeDamage(dmg)) {
   720→      const enemyX = (enemyGo as unknown as { x?: number }).x;
   721→      if (typeof enemyX === 'number' && this.player.sprite?.active) {
   722→        const dir = this.player.sprite.x < enemyX ? -1 : 1;
   723→        this.player.sprite.setVelocityX(dir * 4);
   724→        this.player.sprite.setVelocityY(-4);
   725→      }
   726→    }

   730→    const area = WorldSystem.getCurrentArea();

   733→    for (const npc of npcs) {

   735→      if (dist < 80) {

   739→          return;
   740→        }

   745→  private updatePlay(deltaMs: number): void {
   746→    if (!this.player) return;
   747→    this.player.update(deltaMs);
   748→    this.render.update(this.time.now);
   749→    this.hud?.update();
   750→    // Projectiles
   751→    for (let i = this.projectiles.length - 1; i >= 0; i--) {
   752→      this.projectiles[i].update();
   753→      if (!this.projectiles[i].isAlive) this.projectiles.splice(i, 1);
   754→    }
   755→    // Enemies
   756→    const playerPos = this.player.position;
   757→    for (let i = this.enemies.length - 1; i >= 0; i--) {
   758→      const e = this.enemies[i];
   759→      if (!e.isAlive || !e.sprite || !e.sprite.active) { this.enemies.splice(i, 1); continue; }
   760→      try { e.update(deltaMs, playerPos); } catch { this.enemies.splice(i, 1); continue; }
   761→    }
   762→    // Boss
   763→    if (this.boss && this.boss.isAlive && this.boss.sprite && this.boss.sprite.active) {
   764→      try { this.boss.update(deltaMs); } catch { /* */ }
   765→    }
   766→    // Out of bounds
   767→    if (this.player.sprite.y > GAME.HEIGHT + 80) {
   768→      this.player.takeDamage(25);
   769→      const area = WorldSystem.getCurrentArea();
   770→      if (area) {
   771→        const sec = area.sections.find(s => s.id === this.currentSection);
   772→        if (sec) {
   773→          this.player.sprite.setPosition(sec.x + 200, GAME.HEIGHT - 300);
   774→          this.player.sprite.setVelocity(0, 0);
   775→        }
   776→      }
   777→    }
   778→    // Boss arena zoom
   779→    if (this.bossArenaActive && this.boss && this.boss.isAlive) {
   780→      this.camera.setZoom(0.85);
   781→    }
   782→  }

   799→    this.hud?.destroy();
   800→    this.hud = null;
   801→    this.render?.destroy();
   802→    this.camera.resetZoom();
   803→    this.camera.stopFollow();
   804→    this.camera.setBounds(0, 0, GAME.WIDTH, GAME.HEIGHT);
   805→    this.physicsSys.setWorldBounds(GAME.WIDTH, GAME.HEIGHT);

   807→  }

   944→  /** Gamepad + keyboard navigation for menu/hub/gameover/victory. */
   945→  private handleMenuGamepadNav(input: import('../../systems/InputSystem').InputState): void {
   946→    if (this.menuButtons.length === 0) return;
   947→    this.menuNavCooldown -= 16;
   948→    if (this.menuNavCooldown > 0) return;
   949→
   950→    if (input.leftStickY < -0.3 || input.heldUp) {
   951→      this.menuFocusIndex = (this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length;
   952→      this.updateMenuFocus(); AudioSystem.play('uiHover');
   953→      this.menuNavCooldown = 110;  // freer — was 180
   954→    } else if (input.leftStickY > 0.3 || input.heldDown) {
   955→      this.menuFocusIndex = (this.menuFocusIndex + 1) % this.menuButtons.length;
   956→      this.updateMenuFocus(); AudioSystem.play('uiHover');
   957→      this.menuNavCooldown = 110;
   958→    }
   959→    if (input.jumpPressed || input.firePressed) {
   960→      AudioSystem.play('uiClick');

  1021→  private setupMenuNav(): void {
  1022→    this.menuNavHandler = (e: KeyboardEvent) => {
  1023→      if (this.menuButtons.length === 0) return;
  1024→      if (e.code === 'ArrowUp' || e.code === 'KeyW') {
  1025→        this.menuFocusIndex = (this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length;
  1026→        this.updateMenuFocus(); AudioSystem.play('uiHover');
  1027→      } else if (e.code === 'ArrowDown' || e.code === 'KeyS') {
  1028→        this.menuFocusIndex = (this.menuFocusIndex + 1) % this.menuButtons.length;
  1029→        this.updateMenuFocus(); AudioSystem.play('uiHover');
  1030→      } else if (e.code === 'Enter' || e.code === 'Space') {
  1031→        AudioSystem.play('uiClick');
  1032→        this.menuButtons[this.menuFocusIndex]?.onSelect();
  1033→      }
  1034→    };
  1035→    window.addEventListener('keydown', this.menuNavHandler);
  1036→    this.updateMenuFocus();
  1037→  }

  1039→  private updateMenuFocus(): void {
  1040→    this.menuButtons.forEach((btn, i) => {

  1043→        btn.bg.setStrokeStyle(2, 0x39d0d8, 0.9);
  1044→        btn.bg.setScale(1.05);
  1045→        btn.text.setColor('#66f0ff');
  1046→      } else {
  1047→        btn.bg.setFillStyle(0x0a1018, 0.9);
  1048→        btn.bg.setStrokeStyle(1, 0x1a3040, 0.8);
  1049→        btn.bg.setScale(1);
  1050→        btn.text.setColor('#5a6470');
  1051→      }
  1052→    });
  1053→  }

  1057→  private scheduleDelayed(delay: number, callback: () => void): Phaser.Time.TimerEvent {
  1058→    const timer = this.time.delayedCall(delay, () => {
  1059→      const idx = this.sequenceTimers.indexOf(timer);
  1060→      if (idx >= 0) this.sequenceTimers.splice(idx, 1);
  1061→      callback();
  1062→    });
  1063→    this.sequenceTimers.push(timer);
  1064→    return timer;
  1065→  }
