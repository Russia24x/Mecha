   275тЖТ   275тЖТ    if (this.state === 'play') {
   276тЖТ   276тЖТ      // ESC / Start = toggle pause

   281тЖТ   281тЖТ        if (input.interactPressed) this.tryInteract();
   282тЖТ   282тЖТ        this.updatePlay(deltaMs);
   283тЖТ   283тЖТ      } else {
   284тЖТ   284тЖТ        // Paused тАФ handle pause menu navigation
   285тЖТ   285тЖТ        this.pauseMenuUI.handleNavigation();
   286тЖТ   286тЖТ      }
   287тЖТ   287тЖТ    } else if (this.state === 'menu' || this.state === 'hub' || this.state === 'gameover' || this.state === 'victory') {
   288тЖТ   288тЖТ      // Gamepad + keyboard navigation
   289тЖТ   289тЖТ      this.handleMenuGamepadNav(input);
   290тЖТ   290тЖТ      // ESC in hub = back to menu
   291тЖТ   291тЖТ      if (this.state === 'hub' && input.pausePressed) {
   292тЖТ   292тЖТ        this.setState('menu');
   293тЖТ   293тЖТ      }
   294тЖТ   294тЖТ    }
   295тЖТ   295тЖТ
   296тЖТ   296тЖТ    this.handleDialogueInput(input);
   297тЖТ   297тЖТ  }

   309тЖТ   309тЖТ  private buildMenu(): void {
   310тЖТ   310тЖТ    const c = this.stateContainer!;
   311тЖТ   311тЖТ    const w = GAME.WIDTH, h = GAME.HEIGHT;
   312тЖТ   312тЖТ
   313тЖТ   313тЖТ    // === Background: starry night sky ===
   314тЖТ   314тЖТ    const bg = this.add.graphics();
   315тЖТ   315тЖТ    bg.setDepth(0);
   316тЖТ   316тЖТ    bg.fillStyle(0x040814, 1);
   317тЖТ   317тЖТ    bg.fillRect(0, 0, w, h);
   318тЖТ   318тЖТ    for (let r = 500; r > 0; r -= 30) {
   319тЖТ   319тЖТ      bg.fillStyle(0x0a1228, 0.02);
   320тЖТ   320тЖТ      bg.fillCircle(w / 2, h * 0.35, r);
   321тЖТ   321тЖТ    }
   322тЖТ   322тЖТ    c.add(bg);
   323тЖТ   323тЖТ
   324тЖТ   324тЖТ    // Stars тАФ 120 twinkling, drifting dots
   325тЖТ   325тЖТ    for (let i = 0; i < 120; i++) {
   326тЖТ   326тЖТ      const sx = Math.random() * w;
   327тЖТ   327тЖТ      const sy = Math.random() * h * 0.75;
   328тЖТ   328тЖТ      const size = 0.5 + Math.random() * 2;
   329тЖТ   329тЖТ      const brightness = 0.2 + Math.random() * 0.8;
   330тЖТ   330тЖТ      const starColor = Math.random() < 0.15 ? 0xffe0a0 : Math.random() < 0.3 ? 0xa0c0ff : 0xc0e0ff;
   331тЖТ   331тЖТ      const star = this.add.circle(sx, sy, size, starColor, brightness);
   332тЖТ   332тЖТ      star.setDepth(1);
   333тЖТ   333тЖТ      c.add(star);
   334тЖТ   334тЖТ      this.tweens.add({

   665тЖТ   665тЖТ  private onCollisionStart = (event: MatterJS.IEventCollision<MatterJS.Body>): void => {
   666тЖТ   666тЖТ    for (const pair of event.pairs) {
   667тЖТ   667тЖТ      const aGo = (pair.bodyA as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   668тЖТ   668тЖТ      const bGo = (pair.bodyB as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   669тЖТ   669тЖТ      if (!aGo || !bGo) continue;
   670тЖТ   670тЖТ      const aIsPlayer = aGo.getData('entityType') === 'player';
   671тЖТ   671тЖТ      const bIsPlayer = bGo.getData('entityType') === 'player';
   672тЖТ   672тЖТ      const aSection = aGo.getData('sectionId') as number | undefined;
   673тЖТ   673тЖТ      const bSection = bGo.getData('sectionId') as number | undefined;
   674тЖТ   674тЖТ      if (aIsPlayer && bSection) { this.enterSection(bSection); }
   675тЖТ   675тЖТ      else if (bIsPlayer && aSection) { this.enterSection(aSection); }
   676тЖТ   676тЖТ      else if (aIsPlayer && bGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   677тЖТ   677тЖТ      else if (bIsPlayer && aGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   678тЖТ   678тЖТ      else if (aIsPlayer && bGo.getData('isBossEntry')) { this.enterBossArena(); }
   679тЖТ   679тЖТ      else if (bIsPlayer && aGo.getData('isBossEntry')) { this.enterBossArena(); }
   680тЖТ   680тЖТ      if (aIsPlayer && bGo.getData('entityType') === 'enemy') { this.handleEnemyContact(bGo); }
   681тЖТ   681тЖТ      else if (bIsPlayer && aGo.getData('entityType') === 'enemy') { this.handleEnemyContact(aGo); }
   682тЖТ   682тЖТ      if (aIsPlayer && bGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   683тЖТ   683тЖТ      else if (bIsPlayer && aGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   684тЖТ   684тЖТ    }
   685тЖТ   685тЖТ  };

   745тЖТ   745тЖТ  private updatePlay(deltaMs: number): void {
   746тЖТ   746тЖТ    if (!this.player) return;
   747тЖТ   747тЖТ    this.player.update(deltaMs);
   748тЖТ   748тЖТ    this.render.update(this.time.now);
   749тЖТ   749тЖТ    this.hud?.update();
   750тЖТ   750тЖТ    // Projectiles
   751тЖТ   751тЖТ    for (let i = this.projectiles.length - 1; i >= 0; i--) {
   752тЖТ   752тЖТ      this.projectiles[i].update();
   753тЖТ   753тЖТ      if (!this.projectiles[i].isAlive) this.projectiles.splice(i, 1);
   754тЖТ   754тЖТ    }
   755тЖТ   755тЖТ    // Enemies
   756тЖТ   756тЖТ    const playerPos = this.player.position;
   757тЖТ   757тЖТ    for (let i = this.enemies.length - 1; i >= 0; i--) {
   758тЖТ   758тЖТ      const e = this.enemies[i];
   759тЖТ   759тЖТ      if (!e.isAlive || !e.sprite || !e.sprite.active) { this.enemies.splice(i, 1); continue; }
   760тЖТ   760тЖТ      try { e.update(deltaMs, playerPos); } catch { this.enemies.splice(i, 1); continue; }
   761тЖТ   761тЖТ    }
   762тЖТ   762тЖТ    // Boss
   763тЖТ   763тЖТ    if (this.boss && this.boss.isAlive && this.boss.sprite && this.boss.sprite.active) {
   764тЖТ   764тЖТ      try { this.boss.update(deltaMs); } catch { /* */ }
   765тЖТ   765тЖТ    }
   766тЖТ   766тЖТ    // Out of bounds
   767тЖТ   767тЖТ    if (this.player.sprite.y > GAME.HEIGHT + 80) {
   768тЖТ   768тЖТ      this.player.takeDamage(25);
   769тЖТ   769тЖТ      const area = WorldSystem.getCurrentArea();
   770тЖТ   770тЖТ      if (area) {
   771тЖТ   771тЖТ        const sec = area.sections.find(s => s.id === this.currentSection);
   772тЖТ   772тЖТ        if (sec) {
   773тЖТ   773тЖТ          this.player.sprite.setPosition(sec.x + 200, GAME.HEIGHT - 300);
   774тЖТ   774тЖТ          this.player.sprite.setVelocity(0, 0);
   775тЖТ   775тЖТ        }
   776тЖТ   776тЖТ      }
   777тЖТ   777тЖТ    }
   778тЖТ   778тЖТ    // Boss arena zoom
   779тЖТ   779тЖТ    if (this.bossArenaActive && this.boss && this.boss.isAlive) {
   780тЖТ   780тЖТ      this.camera.setZoom(0.85);
   781тЖТ   781тЖТ    }
   782тЖТ   782тЖТ  }

  1023тЖТ  1023тЖТ      if (this.menuButtons.length === 0) return;

  1031тЖТ  1031тЖТ        AudioSystem.play('uiClick');
  1032тЖТ  1032тЖТ        this.menuButtons[this.menuFocusIndex]?.onSelect();
