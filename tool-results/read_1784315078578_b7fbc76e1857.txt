   198→  /**
   199→   * Compute effective stats from base + unlocked skills.
   200→   * This is the core of the data-driven skill system.
   201→   */
   202→  private computeStats(unlockedSkills: string[]): PlayerStats {
   203→    const base: PlayerStats = {
   204→      maxHealth: PLAYER.MAX_HEALTH,
   205→      maxEnergy: PLAYER.MAX_ENERGY,
   206→      energyRegen: PLAYER.ENERGY_REGEN,
   207→      moveSpeed: PLAYER.MOVE_SPEED,
   208→      jumpVelocity: PLAYER.JUMP_VELOCITY,
   209→      dashSpeed: PLAYER.DASH_SPEED,
   210→      dashDurationMs: PLAYER.DASH_DURATION_MS,
   211→      dashCooldownMs: PLAYER.DASH_COOLDOWN_MS,
   212→      meleeDamage: PLAYER.MELEE_DAMAGE,
   213→      meleeRange: PLAYER.MELEE_RANGE,
   214→      fireCooldownMs: PLAYER.FIRE_COOLDOWN_MS,
   215→      invulnMs: PLAYER.INVULN_MS,
   216→    };
   217→
   218→    // Apply each unlocked skill's effect (existing logic — unchanged)
   219→    for (const skillId of unlockedSkills) {
   220→      const skill = getSkill(skillId);
   221→      if (!skill) continue;
   222→      const eff = skill.effect;
   223→      const statKey = eff.stat as keyof PlayerStats;
   224→      if (eff.multiplier && typeof base[statKey] === 'number') {
   225→        (base[statKey] as number) *= eff.multiplier;
   226→      }
   227→      if (eff.additive && typeof base[statKey] === 'number') {
   228→        (base[statKey] as number) += eff.additive;
   229→      }
   230→    }
   231→
   232→    // ── Apply chassis multipliers (multiplicative layer — order doesn't matter) ──
   233→    // All multipliers are applied after skill effects, so they stack multiplicatively:
   234→    //   final = base × skillMult × chassisMult
   235→    const chassis = getChassis(SaveSystem.getPlayer().selectedChassis as ChassisId);
   236→    const m = chassis.movement;
   237→    const c = chassis.combat;
   238→    base.moveSpeed      *= m.speedMult;
   239→    base.jumpVelocity   *= m.jumpMult;
   240→    base.dashSpeed      *= m.dashMult;
   241→    base.dashCooldownMs *= m.dashCooldownMult;
   242→    base.meleeDamage    *= c.meleeMult;
   243→    base.maxHealth      *= c.maxHealthMult;
   244→    base.maxEnergy      *= c.maxEnergyMult;
   245→    // fireRateMult is "fire rate" (shots per second), so higher = faster firing.
   246→    // fireCooldownMs is the inverse (time between shots), so we divide:
   247→    //   Titan fireRateMult=0.90 → cooldown / 0.90 = longer cooldown = slower firing ✓
   248→    //   Scout fireRateMult=1.10 → cooldown / 1.10 = shorter cooldown = faster firing ✓
   249→    base.fireCooldownMs /= c.fireRateMult;
   250→
   251→    return base;
   252→  }

   254→  /** Rebuild stats when a skill is unlocked. */
   255→  public refreshStats(): void {
   256→    const save = SaveSystem.getPlayer();
   257→    const oldMaxHp = this.stats.maxHealth;
   258→    const oldMaxEn = this.stats.maxEnergy;
   259→    this.stats = this.computeStats(save.unlockedSkills);
   260→    this.abilities = new Set(save.abilities);
   261→    this.maxJumps = this.abilities.has('doubleJump') ? 2 : 1;
   262→    // Adjust current health/energy to new max
   263→    const hpDiff = this.stats.maxHealth - oldMaxHp;
   264→    const enDiff = this.stats.maxEnergy - oldMaxEn;
   265→    this.health.max = this.stats.maxHealth;
   266→    this.energy.max = this.stats.maxEnergy;
   267→    this.energy.regenPerSec = this.stats.energyRegen;
   268→    if (hpDiff > 0) this.health.current += hpDiff;
   269→    if (enDiff > 0) this.energy.current += enDiff;
   270→    this.health.current = Math.min(this.health.current, this.health.max);
   271→    this.energy.current = Math.min(this.energy.current, this.energy.max);
   272→  }

   296→  /** Spawn a pooled particle at (x, y) with given color, radius, life. */
   297→  private spawnPooled(x: number, y: number, radius: number, color: number, lifeMs: number, depth: number = 13): void {
   298→    const p = this.particlePool.find(p => p && !p.visible);
   299→    if (!p) return;
   300→    p.setPosition(x, y);
   301→    p.setRadius(radius);
   302→    p.setFillStyle(color, 0.8);
   303→    p.setDepth(depth);
   304→    p.setAlpha(0.8);
   305→    p.setVisible(true);
   306→    this.scene.tweens.add({
   307→      targets: p, alpha: 0, scale: 0.3, duration: lifeMs,
   308→      onComplete: () => { p.setVisible(false); p.setScale(1); },
   309→    });
   310→  }

   324→  takeDamage(amount: number): boolean {
   325→    if (!this.alive || this.isInvulnerable || amount <= 0) return false;
   326→    this.health.current = Math.max(0, this.health.current - amount);
   327→    this.invulnUntil = this.scene.time.now + this.stats.invulnMs;
   328→    this.flashTimer = this.stats.invulnMs;
   329→    AudioSystem.play('hit');
   330→    EventBus.emit('PLAYER_DAMAGED', { amount, x: this.sprite.x, y: this.sprite.y });
   331→    if (this.health.current <= 0) {
   332→      this.alive = false;
   333→      AudioSystem.play('playerDeath');
   334→      EventBus.emit('PLAYER_DEAD', { id: this.id });
   335→    }
   336→    return true;
   337→  }

   366→  private switchWeapon(dir: 1 | -1): void {
   367→    const list = Object.keys(getWeapon(this.currentWeapon) ? [this.currentWeapon] : []) as WeaponId[];
   368→    const allWeapons = Array.from(this.unlockedWeapons);
   369→    const idx = allWeapons.indexOf(this.currentWeapon);
   370→    const next = allWeapons[(idx + dir + allWeapons.length) % allWeapons.length];
   371→    this.setWeapon(next);
   372→  }

   374→  update(deltaMs: number): void {
   375→    if (!this.alive) return;
   376→    this.updateMovement();
   377→    // ── Abilities ──
   378→    this.tryHover(deltaMs);
   379→    this.updateGrapple(deltaMs);
   380→    this.tryHack(deltaMs);
   381→    this.updateAnimation(deltaMs);
   382→    const gain = this.energy.regenPerSec * (deltaMs / 1000);
   383→    if (gain > 0 && this.energy.current < this.energy.max) {
   384→      this.energy.current = Math.min(this.energy.max, this.energy.current + gain);
   385→    }
   386→  }

   388→  private updateMovement(): void {
   389→    const now = this.scene.time.now;
   390→    const input = InputSystem.getState();
   391→
   392→    // Ground check
   393→    this.grounded = this.physics.isGrounded(this.sprite, PLAYER.BODY_RADIUS);
   394→    if (this.grounded) {
   395→      this.coyoteUntil = now + PLAYER.COYOTE_TIME_MS;
   396→      this.jumpsRemaining = this.maxJumps;
   397→      this.wallJumpUntil = 0;
   398→      if (now < this.jumpBufferUntil) {
   399→        this.sprite.setVelocityY(this.stats.jumpVelocity);
   400→        this.jumpBufferUntil = 0;
   401→        this.grounded = false;
   402→        this.jumpsRemaining--;
   403→        AudioSystem.play('jump');
   404→      }
   405→    }
   406→
   407→    // Wall detection (for wall jump + wall slide) — Mag-Clamp Thrusters
   408→    this.touchingWall = null;
   409→    this.wallSlideActive = false;
   410→    if (!this.grounded && this.abilities.has('wallJump')) {
   411→      const r = PLAYER.BODY_RADIUS;
   412→      const px = this.sprite.x;
   413→      const py = this.sprite.y;
   414→      // Check left wall
   415→      const leftHits = this.physics.bodiesAtPoint(px - r - 4, py);
   416→      if (leftHits.some(b => b.label.startsWith('solid'))) this.touchingWall = 'left';
   417→      // Check right wall
   418→      if (!this.touchingWall) {
   419→        const rightHits = this.physics.bodiesAtPoint(px + r + 4, py);
   420→        if (rightHits.some(b => b.label.startsWith('solid'))) this.touchingWall = 'right';
   421→      }
   422→      // Wall Slide — Mag-Clamp Thrusters: slow descent with sparks + metal sound
   423→      if (this.touchingWall) {
   424→        const pressingToward = (this.touchingWall === 'left' && input.heldLeft) || (this.touchingWall === 'right' && input.heldRight);
   425→        if (pressingToward) {
   426→          const body = this.sprite.body as MatterJS.BodyType;
   427→          if (body && body.velocity.y > 1) {
   428→            this.sprite.setVelocityY(body.velocity.y * 0.45);  // 55% reduction
   429→            this.wallSlideActive = true;
   430→            // Spark + sound effects every 80ms
   431→            if (now - this.lastWallSlideFx >= 80) {
   432→              this.lastWallSlideFx = now;
   433→              const sparkX = px + (this.touchingWall === 'left' ? -r - 2 : r + 2);
   434→              const sparkY = py + (Math.random() - 0.5) * 20;
   435→              this.particles.sparks(sparkX, sparkY, 0xffaa30, 2);
   436→              // Thruster glow
   437→              const glow = this.scene.add.circle(sparkX, sparkY + 10, 4, 0xffc040, 0.4);
   438→              glow.setBlendMode(Phaser.BlendModes.ADD).setDepth(13);
   439→              this.scene.tweens.add({
   440→                targets: glow, alpha: 0, scale: 0.3, duration: 200,
   441→                onComplete: () => glow.destroy(),
   442→              });
   443→            }
   444→          }
   445→        }
   446→      }
   447→    }
   448→
   449→    // Horizontal movement
   450→    let moveX = 0;
   451→    if (input.heldLeft) moveX -= 1;
   452→    if (input.heldRight) moveX += 1;
   453→    if (moveX === 0 && Math.abs(input.leftStickX) > 0.1) moveX = input.leftStickX;
   454→
   455→    // Wall jump lockout — brief horizontal control lock after wall jump
   456→    if (now < this.wallJumpUntil) {
   457→      moveX = 0;
   458→    }
   459→
   460→    if (this.isDashing) {
   461→      const dirSign = this.dashDir === 'right' ? 1 : -1;
   462→      this.sprite.setVelocityX(dirSign * this.stats.dashSpeed);
   463→      const body = this.sprite.body;
   464→      if (body) this.sprite.setVelocityY(body.velocity.y * 0.4);
   465→      if (now - this.lastAfterimageAt >= 35) {
   466→        const pos = this.position;
   467→        const facing = this.dashDir === 'right' ? 1 : -1;
   468→        this.particles.afterimage(pos.x, pos.y - 6, 36, 30, COLORS.PLAYER_GLOW, facing);
   469→        this.lastAfterimageAt = now;
   470→      }
   471→      if (now >= this.dashUntil) this.isDashing = false;
   472→    } else if (now < this.meleeCommitUntil || now < this.fireCommitUntil) {
   473→      // ── Phase 3: Animation commitment — movement locked during attack ──
   474→      // Player can't run while attacking. Decelerate rapidly but allow falling.
   475→      // This makes attacks feel committed and heavy (per Design Pillars: Heavy·Precise·Punishing).
   476→      const body = this.sprite.body;
   477→      this.sprite.setVelocityX((body ? body.velocity.x : 0) * 0.5);
   478→    } else {
   479→      if (moveX !== 0) {
   480→        const targetVx = moveX * this.stats.moveSpeed;
   481→        const body = this.sprite.body;
   482→        const currentVx = body ? body.velocity.x : 0;
   483→        this.sprite.setVelocityX(Phaser.Math.Linear(currentVx, targetVx, 0.35));
   484→        this.facing = moveX > 0 ? 'right' : 'left';
   485→      } else {
   486→        const body = this.sprite.body;
   487→        this.sprite.setVelocityX((body ? body.velocity.x : 0) * 0.78);
   488→      }
   489→    }
   490→
   491→    // Keyboard aim
   492→    let aimY = 0;
   493→    if (input.heldUp) aimY -= 1;
   494→    if (input.heldDown) aimY += 1;
   495→    this.keyAimY = Phaser.Math.Clamp(aimY, -1, 1);
   496→
   497→    // Continuous fire
   498→    if (input.heldFire) this.tryFire();
   499→  }

   501→  private tryJump(): void {
   502→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   503→    this.jumpBufferUntil = this.scene.time.now + PLAYER.JUMP_BUFFER_MS;
   504→    const now = this.scene.time.now;
   505→    // Jump anticipation — squash for 80ms before launch
   506→    this.jumpAnticipationUntil = now + 80;
   507→    this.squashY = 0.85; this.squashX = 1.15;
   508→
   509→    // Wall Jump — Mag-Clamp Thruster burst: powerful push from wall
   510→    if (!this.grounded && this.touchingWall && this.abilities.has('wallJump')) {
   511→      const jumpDir = this.touchingWall === 'left' ? 1 : -1;
   512→      // Thruster burst — powerful launch
   513→      this.sprite.setVelocityY(this.stats.jumpVelocity * 0.95);
   514→      this.sprite.setVelocityX(jumpDir * this.stats.moveSpeed * 1.8);
   515→      this.wallJumpUntil = now + 200;
   516→      this.facing = jumpDir > 0 ? 'right' : 'left';
   517→      this.jumpBufferUntil = 0;
   518→      this.jumpsRemaining = this.maxJumps - 1;
   519→      AudioSystem.play('dash');  // thruster sound
   520→      // Thruster explosion VFX — burst at wall contact point
   521→      const pos = this.position;
   522→      const burstX = pos.x + (this.touchingWall === 'left' ? -PLAYER.BODY_RADIUS : PLAYER.BODY_RADIUS);
   523→      const burstY = pos.y;
   524→      // Ring explosion
   525→      const ring = this.scene.add.circle(burstX, burstY, 6, 0xffc040, 0.8);
   526→      ring.setBlendMode(Phaser.BlendModes.ADD).setDepth(14);
   527→      this.scene.tweens.add({
   528→        targets: ring, scale: { from: 1, to: 3 }, alpha: 0,
   529→        duration: 250, ease: 'Cubic.out', onComplete: () => ring.destroy(),
   530→      });
   531→      // Directional sparks
   532→      this.particles.sparks(burstX, burstY, 0xffaa30, 6);
   533→      // Afterimage for momentum feel
   534→      this.particles.afterimage(pos.x, pos.y - 6, 36, 30, 0x39d0d8, jumpDir);
   535→      return;
   536→    }
   537→
   538→    // Normal jump / double jump
   539→    if (this.jumpsRemaining > 0 && (this.grounded || now < this.coyoteUntil || this.jumpsRemaining > 1)) {
   540→      const isDouble = !this.grounded && this.jumpsRemaining === 1;
   541→      this.sprite.setVelocityY(this.stats.jumpVelocity);
   542→      this.jumpBufferUntil = 0;
   543→      this.grounded = false;
   544→      this.coyoteUntil = 0;
   545→      this.jumpsRemaining--;
   546→      AudioSystem.play(isDouble ? 'doubleJump' : 'jump');
   547→    }
   548→  }

   550→  private tryDash(dir: Direction): void {
   551→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   552→    const now = this.scene.time.now;
   553→    if (this.isDashing || now < this.dashAvailableAt) return;
   554→    if (!this.consumeEnergy(PLAYER.DASH_COST)) return;
   555→    this.isDashing = true;
   556→    this.dashDir = dir;
   557→    this.dashUntil = now + this.stats.dashDurationMs;
   558→    this.dashAvailableAt = now + this.stats.dashDurationMs + this.stats.dashCooldownMs;
   559→    this.invulnUntil = Math.max(this.invulnUntil, this.dashUntil);
   560→    this.lastAfterimageAt = 0;
   561→    AudioSystem.play('dash');
   562→  }

   564→  private getAimDirection(): Phaser.Math.Vector2 {
   565→    const input = InputSystem.getState();
   566→    if (Math.abs(input.rightStickX) > 0.15 || Math.abs(input.rightStickY) > 0.15) {
   567→      const len = Math.sqrt(input.rightStickX ** 2 + input.rightStickY ** 2);
   568→      return new Phaser.Math.Vector2(input.rightStickX / len, input.rightStickY / len);
   569→    }
   570→    const dirX = this.facing === 'right' ? 1 : -1;
   571→    const dirY = this.keyAimY;
   572→    const len = Math.sqrt(dirX * dirX + dirY * dirY);
   573→    return new Phaser.Math.Vector2(dirX / len, dirY / len);
   574→  }

   576→  private tryFire(): void {
   577→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   578→    const now = this.scene.time.now;
   579→    const weapon = getWeapon(this.currentWeapon);
   580→    const cooldown = this.currentWeapon === 'assault_rifle' ? this.stats.fireCooldownMs : weapon.fireRateMs;
   581→    if (now - this.lastFireAt < cooldown) return;
   582→    const energyCost = this.currentWeapon === 'assault_rifle' ? PLAYER.FIRE_COST : weapon.energyCost;
   583→    if (!this.consumeEnergy(energyCost)) return;
   584→    this.lastFireAt = now;
   585→    // ── Phase 3: Animation commitment — lock movement for 80ms after firing ──
   586→    this.fireCommitUntil = now + PlayerEntity.FIRE_COMMIT_MS;
   587→    const from = this.position;
   588→    const aimDir = this.getAimDirection();
   589→    // Muzzle position: follow the gun barrel direction.
   590→    // Gun origin is at player center + y offset (-10 for shoulder height).
   591→    // Barrel is ~28px long, so muzzle is 34px from center in aim direction.
   592→    // Using aimDir * distance ensures bullets always spawn at the gun muzzle
   593→    // regardless of container flip (facing) or rotation.
   594→    const muzzleDist = 34;
   595→    const muzzle = new Phaser.Math.Vector2(
   596→      from.x + aimDir.x * muzzleDist,
   597→      from.y - 10 + aimDir.y * muzzleDist
   598→    );
   599→    AudioSystem.play('fire');
   600→
   601→    // Hitscan weapons (railgun, laser)
   602→    if (weapon.tier === 'hitscan') {
   603→      this.fireHitscan(weapon, muzzle, aimDir);
   604→      return;
   605→    }
   606→
   607→    // Projectile weapons
   608→    const bulletCount = weapon.bulletsPerShot ?? 1;
   609→    for (let i = 0; i < bulletCount; i++) {
   610→      let angleOffset = 0;
   611→      if (bulletCount > 1 && weapon.spread) {
   612→        const t = i / (bulletCount - 1) - 0.5;
   613→        angleOffset = t * weapon.spread;
   614→      }
   615→      const angle = Math.atan2(aimDir.y, aimDir.x) + angleOffset;
   616→      const dir = new Phaser.Math.Vector2(Math.cos(angle), Math.sin(angle));
   617→      const proj = new Projectile(this.scene, this.physics, this.particles, muzzle, dir, {
   618→        speed: weapon.bulletSpeed ?? PLAYER.BULLET_SPEED,
   619→        damage: this.currentWeapon === 'assault_rifle' ? PLAYER.BULLET_DAMAGE : weapon.damage,
   620→        ttl: 1500, owner: 'player', color: weapon.color, size: weapon.size,
   621→        weapon, explosive: weapon.tier === 'explosive', explosionRadius: weapon.explosionRadius,
   622→      });
   623→      this.projectiles.push(proj);
   624→    }
   625→    this.particles.sparks(muzzle.x, muzzle.y, weapon.color, (weapon.bulletsPerShot ?? 1) + 2);
   626→  }

   628→  private fireHitscan(weapon: WeaponData, muzzle: Phaser.Math.Vector2, aimDir: Phaser.Math.Vector2): void {
   629→    const angle = Math.atan2(aimDir.y, aimDir.x);
   630→    const endX = muzzle.x + Math.cos(angle) * weapon.range;
   631→    const endY = muzzle.y + Math.sin(angle) * weapon.range;
   632→    const hit = this.physics.raycastClosest(muzzle.x, muzzle.y, endX, endY, true);
   633→    const endPoint = hit ? new Phaser.Math.Vector2(hit.position.x, hit.position.y) : new Phaser.Math.Vector2(endX, endY);
   634→    // Draw tracer
   635→    const line = this.scene.add.line(0, 0, muzzle.x, muzzle.y, endPoint.x, endPoint.y, weapon.color, 0.9);
   636→    line.setOrigin(0, 0); line.setLineWidth(3); line.setDepth(30);
   637→    this.scene.tweens.add({ targets: line, alpha: 0, lineWidth: 0.5, duration: 120, onComplete: () => line.destroy() });
   638→    // Damage first hit
   639→    if (hit) {
   640→      const go = (hit as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   641→      const entity = go?.getData('entity') as { takeDamage?: (n: number) => void } | undefined;
   642→      if (entity?.takeDamage) {
   643→        entity.takeDamage(weapon.damage);
   644→        this.particles.sparks(endPoint.x, endPoint.y, weapon.color, 4);
   645→      }
   646→    }
   647→  }

   649→  private tryMelee(): void {
   650→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   651→    const now = this.scene.time.now;
   652→    if (now - this.lastMeleeAt < PLAYER.MELEE_COOLDOWN_MS) return;
   653→    if (!this.consumeEnergy(PLAYER.MELEE_COST)) return;
   654→    this.lastMeleeAt = now;
   655→    // ── Phase 3: Animation commitment — lock movement for 200ms after melee ──
   656→    // This is the key "Heavy" feel: you can't run while swinging. Commit to the attack.
   657→    this.meleeCommitUntil = now + PlayerEntity.MELEE_COMMIT_MS;
   658→    const aimDir = this.getAimDirection();
   659→    const origin = this.position;
   660→    const range = this.stats.meleeRange;
   661→    AudioSystem.play('melee');
   662→    this.combat.spawnSlash(origin.x + aimDir.x * 24, origin.y - 6 + aimDir.y * 24, aimDir.x > 0 ? 1 : -1);
   663→    const cx = origin.x + aimDir.x * (range / 2);
   664→    const cy = origin.y - 6 + aimDir.y * (range / 2);
   665→    const bodies = this.physics.bodiesInCircle(cx, cy, range / 2);
   666→    for (const body of bodies) {
   667→      const go = (body as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   668→      if (!go) continue;
   669→      const entityType = go.getData('entityType') as string | undefined;
   670→      const entity = go.getData('entity') as { id?: string; takeDamage?: (n: number) => void } | undefined;
   671→      if (!entity || !entityType) continue;
   672→      if (entityType === 'enemy' || entityType === 'boss') {
   673→        this.combat.dealDamage({
   674→          amount: this.stats.meleeDamage, type: 'melee', source: this.id,
   675→          target: entity.id ?? 'unknown',
   676→          knockback: { x: aimDir.x * 0.18, y: aimDir.y * 0.05 },
   677→          point: { x: (go as unknown as Phaser.GameObjects.Components.Transform).x, y: (go as unknown as Phaser.GameObjects.Components.Transform).y },
   678→        });
   679→      }
   680→    }
   681→  }

   714→  /**
   715→   * GRAPPLE — fire toward nearest grapple anchor in aim direction.
   716→   * Pulls player toward anchor at high speed. Has cooldown.
   717→   */
   718→  private tryGrapple(): void {
   719→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   720→    if (!this.abilities.has('grapple')) return;
   721→    const now = this.scene.time.now;
   722→    if (now < this.grappleAvailableAt) return;
   723→    if (this.grappling) return;
   724→
   725→    // Find nearest grapple anchor within range + in aim direction
   726→    const pos = this.position;
   727→    const aimDir = this.getAimDirection();
   728→    let bestAnchor: Phaser.Math.Vector2 | null = null;
   729→    let bestScore = -1;
   730→    for (const anchor of this.grappleAnchors) {
   731→      const dx = anchor.x - pos.x;
   732→      const dy = anchor.y - pos.y;
   733→      const dist = Math.sqrt(dx * dx + dy * dy);
   734→      if (dist > PlayerEntity.GRAPPLE_RANGE || dist < 30) continue;
   735→      // Direction toward anchor
   736→      const dirX = dx / dist;
   737→      const dirY = dy / dist;
   738→      // Dot product with aim direction — must be > 0.5 (within ~60° of aim)
   739→      const dot = dirX * aimDir.x + dirY * aimDir.y;
   740→      if (dot < 0.5) continue;
   741→      // Score: closer + more aligned = better
   742→      const score = dot * (1 - dist / PlayerEntity.GRAPPLE_RANGE);
   743→      if (score > bestScore) {
   744→        bestScore = score;
   745→        bestAnchor = anchor;
   746→      }
   747→    }
   748→
   749→    if (!bestAnchor) return;
   750→
   751→    // Start grapple
   752→    this.grappling = true;
   753→    this.grappleAnchor = bestAnchor;
   754→    this.grappleUntil = now + PlayerEntity.GRAPPLE_DURATION_MS;
   755→    this.grappleAvailableAt = now + PlayerEntity.GRAPPLE_DURATION_MS + PlayerEntity.GRAPPLE_COOLDOWN_MS;
   756→    AudioSystem.play('dash');  // reuse dash sound for now
   757→
   758→    // Draw grapple line
   759→    this.grappleLine = this.scene.add.line(0, 0, pos.x, pos.y, bestAnchor.x, bestAnchor.y, 0x66f0ff, 0.8);
   760→    this.grappleLine.setOrigin(0, 0);
   761→    this.grappleLine.setLineWidth(2);
   762→    this.grappleLine.setDepth(20);
   763→  }
   764→
   765→  /** Update grapple state — pull player toward anchor. */
   766→  private updateGrapple(deltaMs: number): void {
   767→    if (!this.grappling || !this.grappleAnchor) return;
   768→    const now = this.scene.time.now;
   769→    if (now >= this.grappleUntil) {
   770→      this.endGrapple();
   771→      return;
   772→    }
   773→    const pos = this.position;
   774→    const dx = this.grappleAnchor.x - pos.x;
   775→    const dy = this.grappleAnchor.y - pos.y;
   776→    const dist = Math.sqrt(dx * dx + dy * dy);
   777→    if (dist < 30) {
   778→      this.endGrapple();
   779→      return;
   780→    }
   781→    // Pull toward anchor
   782→    const vx = (dx / dist) * PlayerEntity.GRAPPLE_SPEED;
   783→    const vy = (dy / dist) * PlayerEntity.GRAPPLE_SPEED;
   784→    this.sprite.setVelocity(vx, vy);
   785→    // Update grapple line
   786→    if (this.grappleLine && this.grappleLine.active) {
   787→      this.grappleLine.setTo(pos.x, pos.y, this.grappleAnchor.x, this.grappleAnchor.y);
   788→    }
   789→  }
   790→
   791→  private endGrapple(): void {
   792→    this.grappling = false;
   793→    this.grappleAnchor = null;
   794→    if (this.grappleLine) {
   795→      this.scene.tweens.add({
   796→        targets: this.grappleLine, alpha: 0, duration: 150,
   797→        onComplete: () => this.grappleLine?.destroy(),
   798→      });
   799→      this.grappleLine = null;
   800→    }
   801→  }

   805→   * Costs 40 energy, 3s cooldown.

   807→  private tryEMP(): void {

   810→    const now = this.scene.time.now;
   811→    if (now < this.empAvailableAt) return;

   813→    this.empAvailableAt = now + PlayerEntity.EMP_COOLDOWN_MS;

   815→
   816→    const pos = this.position;

   858→    EventBus.emit('EMP_PULSE', { x: pos.x, y: pos.y, radius: PlayerEntity.EMP_RADIUS });
   859→
   860→    // Screen flash
   861→    this.particles.screenFlash(0x66f0ff, 0.25, 300);
   862→    this.scene.cameras.main.shake(150, 0.006);
   863→  }

   865→  /**
   866→   * HACK — hold interact (E / B / Circle) near a hackable enemy to convert it to friendly.
   867→   * Progress bar fills over 1.5s. On success, enemy fights for player.
   868→   * ── FIX Bug 2: Now requires heldInteract to start + continue hack ──
   869→   */
   870→  private tryHack(deltaMs: number): void {
   871→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   872→    if (!this.abilities.has('hack')) { this.cancelHack(); return; }
   873→
   874→    const input = InputSystem.getState();
   875→    const pos = this.position;
   876→
   877→    // Find nearest hackable enemy within range
   878→    let nearestHackable: import('../enemies/EnemyEntity').EnemyEntity | null = null;
   879→    let nearestDist = 60;  // hack range
   880→    for (const enemy of this.enemiesRef) {
   881→      if (!enemy.isAlive) continue;
   882→      if (!enemy.data.hackable) continue;
   883→      const ePos = enemy.position;
   884→      const dist = Phaser.Math.Distance.Between(pos.x, pos.y, ePos.x, ePos.y);
   885→      if (dist < nearestDist) {
   886→        nearestDist = dist;
   887→        nearestHackable = enemy;
   888→      }
   889→    }
   890→
   891→    // No hackable enemy nearby → cancel any in-progress hack
   892→    if (!nearestHackable) {
   893→      this.cancelHack();
   894→      return;
   895→    }
   896→
   897→    // ── FIX Bug 2: Require held interact to start + continue hack ──
   898→    if (!input.heldInteract) {
   899→      // Player not holding interact → cancel any in-progress hack
   900→      this.cancelHack();
   901→      return;
   902→    }
   903→
   904→    // Player is holding interact + near a hackable enemy → start or continue hack
   905→    if (!this.hackTarget || this.hackTarget !== nearestHackable) {
   906→      this.hackTarget = nearestHackable;
   907→      this.hackProgress = 0;
   908→    }
   909→    this.hackProgress += deltaMs;
   910→
   911→    // Show progress bar
   912→    if (!this.hackProgressBar) {
   913→      this.hackProgressBar = this.scene.add.rectangle(0, 0, 40, 4, 0x40ff80, 0.9);
   914→      this.hackProgressBar.setDepth(16);
   915→    }
   916→    const ePos = this.hackTarget.position;
   917→    this.hackProgressBar.setPosition(ePos.x, ePos.y - 30);
   918→    this.hackProgressBar.setDisplaySize(40 * (this.hackProgress / PlayerEntity.HACK_DURATION_MS), 4);
   919→    this.hackProgressBar.setVisible(true);
   920→
   921→    // Hack complete?
   922→    if (this.hackProgress >= PlayerEntity.HACK_DURATION_MS) {
   923→      this.completeHack();
   924→    }
   925→  }

   927→  private completeHack(): void {
   928→    if (!this.hackTarget) return;
   929→    // Convert enemy to friendly — emit event for GameScene to handle
   930→    EventBus.emit('HACK_COMPLETE', { enemyId: this.hackTarget.id });
   931→    AudioSystem.play('skillUnlock');
   932→    this.particles.screenFlash(0x40ff80, 0.2, 300);
   933→    this.cancelHack();
   934→  }
   935→
   936→  private cancelHack(): void {
   937→    this.hackTarget = null;
   938→    this.hackProgress = 0;
   939→    if (this.hackProgressBar) {
   940→      this.hackProgressBar.setVisible(false);
   941→    }
   942→  }
   943→
   944→  private updateAnimation(deltaMs: number): void {
   945→    if (!this.visual) return;
   946→    this.animTime += deltaMs;
   947→    const pos = this.position;
   948→    const facing = this.facing === 'right' ? 1 : -1;
   949→    const body = this.sprite.body;
   950→    const vx = body?.velocity.x ?? 0;
   951→    const vy = body?.velocity.y ?? 0;
   952→    const isMoving = Math.abs(vx) > 0.5 && this.grounded;
   953→    const isJumping = !this.grounded;
   954→    const isDashing = this.isDashing;
   955→
   956→    // ── Physics polish: squash & stretch ──
   957→    // Recover from anticipation
   958→    if (this.scene.time.now > this.jumpAnticipationUntil) {
   959→      // Landing squash — if we just landed with downward velocity
   960→      if (this.grounded && !this.groundedLastFrame && this.lastVy > 4) {
   961→        this.squashY = 0.7; this.squashX = 1.25;
   962→      }
   963→      // Stretch on fast upward movement (jump apex)
   964→      if (isJumping && vy < -6) {
   965→        this.squashY = Phaser.Math.Linear(this.squashY, 1.2, 0.15);
   966→        this.squashX = Phaser.Math.Linear(this.squashX, 0.88, 0.15);
   967→      } else {
   968→        // Ease back to neutral
   969→        this.squashY = Phaser.Math.Linear(this.squashY, 1, 0.18);
   970→        this.squashX = Phaser.Math.Linear(this.squashX, 1, 0.18);
   971→      }
   972→    }
   973→
   974→    // ── Momentum tilt — lean into the direction of motion ──
   975→    const targetTilt = isDashing
   976→      ? facing * 0.25
   977→      : isMoving
   978→        ? facing * (vx / 30)  // subtle lean
   979→        : isJumping
   980→          ? (vy < 0 ? facing * 0.08 : facing * 0.04)  // tilt up when rising, less when falling
   981→          : 0;
   982→    this.tilt = Phaser.Math.Linear(this.tilt, targetTilt, 0.12);
   983→
   984→    // ── Apply transform to visual container ──
   985→    // Restore old body bob: subtle vertical sine when moving (old "mazze" feel)
   986→    const bob = isMoving ? Math.sin(this.animTime / 80) * 2 : 0;
   987→    this.visual.container.setPosition(pos.x, pos.y - 4 + bob);
   988→    this.visual.container.setRotation(this.tilt);
   989→    this.visual.container.setScale(facing * this.squashX, this.squashY);
   990→    this.visual.setFacing(facing as 1 | -1);
   991→
   992→    // ── Restore old leg swing animation (combined with new factory visual) ──
   993→    if (this.visual.setWalkPhase) {
   994→      this.visual.setWalkPhase(this.animTime / 100, isMoving, isJumping);
   995→    }
   996→
   997→    // ── Restore old gun barrel rotation toward aim direction (all-direction aim) ──
   998→    if (this.visual.setGunAngle) {
   999→      const aimDir = this.getAimDirection();
  1000→      const targetAngle = Math.atan2(aimDir.y, aimDir.x);
  1001→      // Smoothly rotate toward target (old interpolation)
  1002→      this.aimAngle = Phaser.Math.Angle.RotateTo(this.aimAngle, targetAngle, 0.3);
  1003→      this.visual.setGunAngle(this.aimAngle);
  1004→    }
  1005→
  1006→    // ── Core pulse — brighten on dash, dim on low energy ──
  1007→    const energyPct = this.energy.current / this.energy.max;
  1008→    const dashBoost = isDashing ? 0.4 : 0;
  1009→    const pulseBrightness = 0.4 + Math.sin(this.animTime / 200) * 0.15 + dashBoost;
  1010→    this.visual.setCorePulse(Math.min(1, pulseBrightness * (0.5 + energyPct * 0.5)));
  1011→
  1012→    // ── Visual tier upgrade — mech evolves as player levels up ──
  1013→    // Tier 0 (Lv1-4): base, Tier 1 (Lv5-9): brighter core, Tier 2 (Lv10-14): shoulder cannons, Tier 3 (Lv15+): crest + big thrusters
  1014→    if (this.visual.setTier) {
  1015→      const level = SaveSystem.getPlayer().level;
  1016→      const tier = level >= 15 ? 3 : level >= 10 ? 2 : level >= 5 ? 1 : 0;
  1017→      if (tier !== this.lastTier) {
  1018→        this.visual.setTier(tier);
  1019→        this.lastTier = tier;
  1020→      }
  1021→    }
  1022→
  1023→    // ── Thruster VFX ──
  1024→    // Intensity 0..1 based on: jumping (full), dashing (full), falling fast (medium), grounded (0)
  1025→    let thrusterIntensity = 0;
  1026→    if (isDashing) thrusterIntensity = 0.9;
  1027→    else if (isJumping && vy < 0) thrusterIntensity = 0.7;  // rising
  1028→    else if (isJumping && vy > 2) thrusterIntensity = 0.3;  // falling gently
  1029→    else if (this.wallSlideActive) thrusterIntensity = 0.5;
  1030→    this.visual.setThrusterIntensity(thrusterIntensity);
  1031→
  1032→    // Emit thruster particles when intensity is high
  1033→    const now = this.scene.time.now;
  1034→    if (thrusterIntensity > 0.4 && now - this.lastThrusterEmit > 30) {
  1035→      this.lastThrusterEmit = now;
  1036→      // Emit downward thruster flame particles (pooled)
  1037→      for (const dx of [-9, 9]) {
  1038→        this.spawnPooled(pos.x + dx * facing, pos.y + 12, 2 + Math.random() * 2, 0xffc040, 150);
  1039→      }
  1040→    }
  1041→
  1042→    // ── Landing impact — dust + camera shake ──
  1043→    if (this.grounded && !this.wasGrounded && this.lastVy > 3) {
  1044→      this.particles.dust(pos.x, pos.y + 20);
  1045→      this.scene.cameras.main.shake(60, 0.003 * Math.min(this.lastVy / 5, 1));
  1046→      // Extra dust puff for hard landings (pooled)
  1047→      if (this.lastVy > 7) {
  1048→        for (let i = 0; i < 6; i++) {
  1049→          const dustX = pos.x + (Math.random() - 0.5) * 30;
  1050→          this.spawnPooled(dustX, pos.y + 20, 3, 0x6a5a4a, 400, 12);
  1051→        }
  1052→      }
  1053→    }
  1054→
  1055→    // ── Per-step dust when running (pooled) ──
  1056→    if (isMoving && now - this.lastThrusterEmit > 100) {
  1057→      this.spawnPooled(pos.x + (Math.random() - 0.5) * 8, pos.y + 18, 1.5, 0x4a3a2a, 250, 11);
  1058→    }
  1059→
  1060→    this.wasGrounded = this.grounded;
  1061→    this.groundedLastFrame = this.grounded;
  1062→    this.lastVy = vy;
  1063→    this.lastVx = vx;
  1064→
  1065→    // ── Flash while invulnerable ──
  1066→    if (this.flashTimer > 0) {
  1067→      this.flashTimer -= deltaMs;
  1068→      const on = Math.floor(this.flashTimer / 80) % 2 === 0;
  1069→      this.visual.container.setAlpha(on ? 0.4 : 1);
  1070→    } else {
  1071→      this.visual.container.setAlpha(1);
  1072→    }
  1073→  }
  1074→
  1075→  destroy(): void {
  1076→    // Clear input callbacks so destroyed player doesn't receive input
  1077→    InputSystem.clearCallbacks();
  1078→    this.visual?.destroy();
  1079→    this.visual = null;
  1080→    // Clean up ability visuals
  1081→    this.grappleLine?.destroy();
  1082→    this.grappleLine = null;
  1083→    this.hackProgressBar?.destroy();
  1084→    this.hackProgressBar = null;
  1085→    if (this.sprite && this.sprite.active) this.sprite.destroy();
  1086→  }
  1087→}
