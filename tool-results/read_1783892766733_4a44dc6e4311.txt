   142→  /**
   143→   * Compute effective stats from base + unlocked skills.
   144→   * This is the core of the data-driven skill system.
   145→   */
   146→  private computeStats(unlockedSkills: string[]): PlayerStats {
   147→    const base: PlayerStats = {
   148→      maxHealth: PLAYER.MAX_HEALTH,
   149→      maxEnergy: PLAYER.MAX_ENERGY,
   150→      energyRegen: PLAYER.ENERGY_REGEN,
   151→      moveSpeed: PLAYER.MOVE_SPEED,
   152→      jumpVelocity: PLAYER.JUMP_VELOCITY,
   153→      dashSpeed: PLAYER.DASH_SPEED,
   154→      dashDurationMs: PLAYER.DASH_DURATION_MS,
   155→      dashCooldownMs: PLAYER.DASH_COOLDOWN_MS,
   156→      meleeDamage: PLAYER.MELEE_DAMAGE,
   157→      meleeRange: PLAYER.MELEE_RANGE,
   158→      fireCooldownMs: PLAYER.FIRE_COOLDOWN_MS,
   159→      invulnMs: PLAYER.INVULN_MS,
   160→    };
   161→
   162→    // Apply each unlocked skill's effect
   163→    for (const skillId of unlockedSkills) {
   164→      const skill = getSkill(skillId);
   165→      if (!skill) continue;
   166→      const eff = skill.effect;
   167→      const statKey = eff.stat as keyof PlayerStats;
   168→      if (eff.multiplier && typeof base[statKey] === 'number') {
   169→        (base[statKey] as number) *= eff.multiplier;
   170→      }
   171→      if (eff.additive && typeof base[statKey] === 'number') {
   172→        (base[statKey] as number) += eff.additive;
   173→      }
   174→    }
   175→
   176→    return base;
   177→  }

   179→  /** Rebuild stats when a skill is unlocked. */
   180→  public refreshStats(): void {
   181→    const save = SaveSystem.getPlayer();
   182→    const oldMaxHp = this.stats.maxHealth;
   183→    const oldMaxEn = this.stats.maxEnergy;
   184→    this.stats = this.computeStats(save.unlockedSkills);
   185→    this.abilities = new Set(save.abilities);
   186→    this.maxJumps = this.abilities.has('doubleJump') ? 2 : 1;
   187→    // Adjust current health/energy to new max
   188→    const hpDiff = this.stats.maxHealth - oldMaxHp;
   189→    const enDiff = this.stats.maxEnergy - oldMaxEn;
   190→    this.health.max = this.stats.maxHealth;
   191→    this.energy.max = this.stats.maxEnergy;
   192→    this.energy.regenPerSec = this.stats.energyRegen;
   193→    if (hpDiff > 0) this.health.current += hpDiff;
   194→    if (enDiff > 0) this.energy.current += enDiff;
   195→    this.health.current = Math.min(this.health.current, this.health.max);
   196→    this.energy.current = Math.min(this.energy.current, this.energy.max);
   197→  }

   227→  takeDamage(amount: number): boolean {
   228→    if (!this.alive || this.isInvulnerable || amount <= 0) return false;
   229→    this.health.current = Math.max(0, this.health.current - amount);
   230→    this.invulnUntil = this.scene.time.now + this.stats.invulnMs;
   231→    this.flashTimer = this.stats.invulnMs;
   232→    AudioSystem.play('hit');
   233→    EventBus.emit('PLAYER_DAMAGED', { amount, x: this.sprite.x, y: this.sprite.y });
   234→    if (this.health.current <= 0) {
   235→      this.alive = false;
   236→      AudioSystem.play('playerDeath');
   237→      EventBus.emit('PLAYER_DEAD', { id: this.id });
   238→    }
   239→    return true;
   240→  }

   242→  heal(amount: number): void {
   243→    this.health.current = Math.min(this.health.max, this.health.current + amount);
   244→    EventBus.emit('PLAYER_DAMAGED', { amount: -amount, x: this.sprite.x, y: this.sprite.y, heal: true });
   245→  }

   269→  private switchWeapon(dir: 1 | -1): void {
   270→    const list = Object.keys(getWeapon(this.currentWeapon) ? [this.currentWeapon] : []) as WeaponId[];
   271→    const allWeapons = Array.from(this.unlockedWeapons);
   272→    const idx = allWeapons.indexOf(this.currentWeapon);
   273→    const next = allWeapons[(idx + dir + allWeapons.length) % allWeapons.length];
   274→    this.setWeapon(next);
   275→  }

   277→  update(deltaMs: number): void {
   278→    if (!this.alive) return;
   279→    this.updateMovement();
   280→    this.updateAnimation(deltaMs);
   281→    const gain = this.energy.regenPerSec * (deltaMs / 1000);
   282→    if (gain > 0 && this.energy.current < this.energy.max) {
   283→      this.energy.current = Math.min(this.energy.max, this.energy.current + gain);
   284→    }
   285→  }

   287→  private updateMovement(): void {
   288→    const now = this.scene.time.now;
   289→    const input = InputSystem.getState();
   290→
   291→    // Ground check
   292→    this.grounded = this.physics.isGrounded(this.sprite, PLAYER.BODY_RADIUS);
   293→    if (this.grounded) {
   294→      this.coyoteUntil = now + PLAYER.COYOTE_TIME_MS;
   295→      this.jumpsRemaining = this.maxJumps;
   296→      this.wallJumpUntil = 0;
   297→      if (now < this.jumpBufferUntil) {
   298→        this.sprite.setVelocityY(this.stats.jumpVelocity);
   299→        this.jumpBufferUntil = 0;
   300→        this.grounded = false;
   301→        this.jumpsRemaining--;
   302→        AudioSystem.play('jump');
   303→      }
   304→    }
   305→
   306→    // Wall detection (for wall jump + wall slide) — Mag-Clamp Thrusters
   307→    this.touchingWall = null;
   308→    this.wallSlideActive = false;
   309→    if (!this.grounded && this.abilities.has('wallJump')) {
   310→      const r = PLAYER.BODY_RADIUS;
   311→      const px = this.sprite.x;
   312→      const py = this.sprite.y;
   313→      // Check left wall
   314→      const leftHits = this.physics.bodiesAtPoint(px - r - 4, py);
   315→      if (leftHits.some(b => b.label.startsWith('solid'))) this.touchingWall = 'left';
   316→      // Check right wall
   317→      if (!this.touchingWall) {
   318→        const rightHits = this.physics.bodiesAtPoint(px + r + 4, py);
   319→        if (rightHits.some(b => b.label.startsWith('solid'))) this.touchingWall = 'right';
   320→      }
   321→      // Wall Slide — Mag-Clamp Thrusters: slow descent with sparks + metal sound
   322→      if (this.touchingWall) {
   323→        const pressingToward = (this.touchingWall === 'left' && input.heldLeft) || (this.touchingWall === 'right' && input.heldRight);
   324→        if (pressingToward) {
   325→          const body = this.sprite.body as MatterJS.BodyType;
   326→          if (body && body.velocity.y > 1) {
   327→            this.sprite.setVelocityY(body.velocity.y * 0.45);  // 55% reduction
   328→            this.wallSlideActive = true;
   329→            // Spark + sound effects every 80ms
   330→            if (now - this.lastWallSlideFx >= 80) {
   331→              this.lastWallSlideFx = now;
   332→              const sparkX = px + (this.touchingWall === 'left' ? -r - 2 : r + 2);
   333→              const sparkY = py + (Math.random() - 0.5) * 20;
   334→              this.particles.sparks(sparkX, sparkY, 0xffaa30, 2);
   335→              // Thruster glow
   336→              const glow = this.scene.add.circle(sparkX, sparkY + 10, 4, 0xffc040, 0.4);
   337→              glow.setBlendMode(Phaser.BlendModes.ADD).setDepth(13);
   338→              this.scene.tweens.add({
   339→                targets: glow, alpha: 0, scale: 0.3, duration: 200,
   340→                onComplete: () => glow.destroy(),
   341→              });
   342→            }
   343→          }
   344→        }
   345→      }
   346→    }
   347→
   348→    // Horizontal movement
   349→    let moveX = 0;
   350→    if (input.heldLeft) moveX -= 1;
   351→    if (input.heldRight) moveX += 1;
   352→    if (moveX === 0 && Math.abs(input.leftStickX) > 0.1) moveX = input.leftStickX;
   353→
   354→    // Wall jump lockout — brief horizontal control lock after wall jump
   355→    if (now < this.wallJumpUntil) {
   356→      moveX = 0;
   357→    }
   358→
   359→    if (this.isDashing) {
   360→      const dirSign = this.dashDir === 'right' ? 1 : -1;
   361→      this.sprite.setVelocityX(dirSign * this.stats.dashSpeed);
   362→      const body = this.sprite.body;
   363→      if (body) this.sprite.setVelocityY(body.velocity.y * 0.4);
   364→      if (now - this.lastAfterimageAt >= 35) {
   365→        const pos = this.position;
   366→        const facing = this.dashDir === 'right' ? 1 : -1;
   367→        this.particles.afterimage(pos.x, pos.y - 6, 36, 30, COLORS.PLAYER_GLOW, facing);
   368→        this.lastAfterimageAt = now;
   369→      }
   370→      if (now >= this.dashUntil) this.isDashing = false;
   371→    } else {
   372→      if (moveX !== 0) {
   373→        const targetVx = moveX * this.stats.moveSpeed;
   374→        const body = this.sprite.body;
   375→        const currentVx = body ? body.velocity.x : 0;
   376→        this.sprite.setVelocityX(Phaser.Math.Linear(currentVx, targetVx, 0.35));
   377→        this.facing = moveX > 0 ? 'right' : 'left';
   378→      } else {
   379→        const body = this.sprite.body;
   380→        this.sprite.setVelocityX((body ? body.velocity.x : 0) * 0.78);
   381→      }
   382→    }
   383→
   384→    // Keyboard aim
   385→    let aimY = 0;
   386→    if (input.heldUp) aimY -= 1;
   387→    if (input.heldDown) aimY += 1;
   388→    this.keyAimY = Phaser.Math.Clamp(aimY, -1, 1);
   389→
   390→    // Continuous fire
   391→    if (input.heldFire) this.tryFire();
   392→  }

   394→  private tryJump(): void {
   395→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   396→    this.jumpBufferUntil = this.scene.time.now + PLAYER.JUMP_BUFFER_MS;
   397→    const now = this.scene.time.now;
   398→
   399→    // Wall Jump — Mag-Clamp Thruster burst: powerful push from wall
   400→    if (!this.grounded && this.touchingWall && this.abilities.has('wallJump')) {
   401→      const jumpDir = this.touchingWall === 'left' ? 1 : -1;
   402→      // Thruster burst — powerful launch
   403→      this.sprite.setVelocityY(this.stats.jumpVelocity * 0.95);
   404→      this.sprite.setVelocityX(jumpDir * this.stats.moveSpeed * 1.8);
   405→      this.wallJumpUntil = now + 200;
   406→      this.facing = jumpDir > 0 ? 'right' : 'left';
   407→      this.jumpBufferUntil = 0;
   408→      this.jumpsRemaining = this.maxJumps - 1;
   409→      AudioSystem.play('dash');  // thruster sound
   410→      // Thruster explosion VFX — burst at wall contact point
   411→      const pos = this.position;
   412→      const burstX = pos.x + (this.touchingWall === 'left' ? -PLAYER.BODY_RADIUS : PLAYER.BODY_RADIUS);
   413→      const burstY = pos.y;
   414→      // Ring explosion
   415→      const ring = this.scene.add.circle(burstX, burstY, 6, 0xffc040, 0.8);
   416→      ring.setBlendMode(Phaser.BlendModes.ADD).setDepth(14);
   417→      this.scene.tweens.add({
   418→        targets: ring, scale: { from: 1, to: 3 }, alpha: 0,
   419→        duration: 250, ease: 'Cubic.out', onComplete: () => ring.destroy(),
   420→      });
   421→      // Directional sparks
   422→      this.particles.sparks(burstX, burstY, 0xffaa30, 6);
   423→      // Afterimage for momentum feel
   424→      this.particles.afterimage(pos.x, pos.y - 6, 36, 30, 0x39d0d8, jumpDir);
   425→      return;
   426→    }
   427→
   428→    // Normal jump / double jump
   429→    if (this.jumpsRemaining > 0 && (this.grounded || now < this.coyoteUntil || this.jumpsRemaining > 1)) {
   430→      const isDouble = !this.grounded && this.jumpsRemaining === 1;
   431→      this.sprite.setVelocityY(this.stats.jumpVelocity);
   432→      this.jumpBufferUntil = 0;
   433→      this.grounded = false;
   434→      this.coyoteUntil = 0;
   435→      this.jumpsRemaining--;
   436→      AudioSystem.play(isDouble ? 'doubleJump' : 'jump');
   437→    }
   438→  }
   439→
   440→  private tryDash(dir: Direction): void {
   441→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   442→    const now = this.scene.time.now;
   443→    if (this.isDashing || now < this.dashAvailableAt) return;
   444→    if (!this.consumeEnergy(PLAYER.DASH_COST)) return;
   445→    this.isDashing = true;
   446→    this.dashDir = dir;
   447→    this.dashUntil = now + this.stats.dashDurationMs;
   448→    this.dashAvailableAt = now + this.stats.dashDurationMs + this.stats.dashCooldownMs;
   449→    this.invulnUntil = Math.max(this.invulnUntil, this.dashUntil);
   450→    this.lastAfterimageAt = 0;
   451→    AudioSystem.play('dash');
   452→  }
   453→
   454→  private getAimDirection(): Phaser.Math.Vector2 {
   455→    const input = InputSystem.getState();
   456→    if (Math.abs(input.rightStickX) > 0.15 || Math.abs(input.rightStickY) > 0.15) {
   457→      const len = Math.sqrt(input.rightStickX ** 2 + input.rightStickY ** 2);
   458→      return new Phaser.Math.Vector2(input.rightStickX / len, input.rightStickY / len);
   459→    }
   460→    const dirX = this.facing === 'right' ? 1 : -1;
   461→    const dirY = this.keyAimY;
   462→    const len = Math.sqrt(dirX * dirX + dirY * dirY);
   463→    return new Phaser.Math.Vector2(dirX / len, dirY / len);
   464→  }
   465→
   466→  private tryFire(): void {
   467→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   468→    const now = this.scene.time.now;
   469→    const weapon = getWeapon(this.currentWeapon);
   470→    const cooldown = this.currentWeapon === 'assault_rifle' ? this.stats.fireCooldownMs : weapon.fireRateMs;
   471→    if (now - this.lastFireAt < cooldown) return;
   472→    const energyCost = this.currentWeapon === 'assault_rifle' ? PLAYER.FIRE_COST : weapon.energyCost;
   473→    if (!this.consumeEnergy(energyCost)) return;
   474→    this.lastFireAt = now;
   475→    const from = this.position;
   476→    const aimDir = this.getAimDirection();
   477→    const muzzle = new Phaser.Math.Vector2(from.x + aimDir.x * 30, from.y - 6 + aimDir.y * 30);
   478→    AudioSystem.play('fire');
   479→
   480→    // Hitscan weapons (railgun, laser)
   481→    if (weapon.tier === 'hitscan') {
   482→      this.fireHitscan(weapon, muzzle, aimDir);
   483→      return;
   484→    }
   485→
   486→    // Projectile weapons
   487→    const bulletCount = weapon.bulletsPerShot ?? 1;
   488→    for (let i = 0; i < bulletCount; i++) {
   489→      let angleOffset = 0;
   490→      if (bulletCount > 1 && weapon.spread) {
   491→        const t = i / (bulletCount - 1) - 0.5;
   492→        angleOffset = t * weapon.spread;
   493→      }
   494→      const angle = Math.atan2(aimDir.y, aimDir.x) + angleOffset;
   495→      const dir = new Phaser.Math.Vector2(Math.cos(angle), Math.sin(angle));
   496→      const proj = new Projectile(this.scene, this.physics, this.particles, muzzle, dir, {
   497→        speed: weapon.bulletSpeed ?? PLAYER.BULLET_SPEED,
   498→        damage: this.currentWeapon === 'assault_rifle' ? PLAYER.BULLET_DAMAGE : weapon.damage,
   499→        ttl: 1500, owner: 'player', color: weapon.color, size: weapon.size,
   500→        weapon, explosive: weapon.tier === 'explosive', explosionRadius: weapon.explosionRadius,
   501→      });
   502→      this.projectiles.push(proj);
   503→    }
   504→    this.particles.sparks(muzzle.x, muzzle.y, weapon.color, (weapon.bulletsPerShot ?? 1) + 2);
   505→  }
   506→
   507→  private fireHitscan(weapon: WeaponData, muzzle: Phaser.Math.Vector2, aimDir: Phaser.Math.Vector2): void {
   508→    const angle = Math.atan2(aimDir.y, aimDir.x);
   509→    const endX = muzzle.x + Math.cos(angle) * weapon.range;
   510→    const endY = muzzle.y + Math.sin(angle) * weapon.range;
   511→    const hit = this.physics.raycastClosest(muzzle.x, muzzle.y, endX, endY, true);
   512→    const endPoint = hit ? new Phaser.Math.Vector2(hit.position.x, hit.position.y) : new Phaser.Math.Vector2(endX, endY);
   513→    // Draw tracer
   514→    const line = this.scene.add.line(0, 0, muzzle.x, muzzle.y, endPoint.x, endPoint.y, weapon.color, 0.9);
   515→    line.setOrigin(0, 0); line.setLineWidth(3); line.setDepth(30);
   516→    this.scene.tweens.add({ targets: line, alpha: 0, lineWidth: 0.5, duration: 120, onComplete: () => line.destroy() });
   517→    // Damage first hit
   518→    if (hit) {
   519→      const go = (hit as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   520→      const entity = go?.getData('entity') as { takeDamage?: (n: number) => void } | undefined;
   521→      if (entity?.takeDamage) {
   522→        entity.takeDamage(weapon.damage);
   523→        this.particles.sparks(endPoint.x, endPoint.y, weapon.color, 4);
   524→      }
   525→    }
   526→  }
   527→
   528→  private tryMelee(): void {
   529→    if (!this.alive || !this.sprite || !this.sprite.active) return;
   530→    const now = this.scene.time.now;
   531→    if (now - this.lastMeleeAt < PLAYER.MELEE_COOLDOWN_MS) return;
   532→    if (!this.consumeEnergy(PLAYER.MELEE_COST)) return;
   533→    this.lastMeleeAt = now;
   534→    const aimDir = this.getAimDirection();
   535→    const origin = this.position;
   536→    const range = this.stats.meleeRange;
   537→    AudioSystem.play('melee');
   538→    this.combat.spawnSlash(origin.x + aimDir.x * 24, origin.y - 6 + aimDir.y * 24, aimDir.x > 0 ? 1 : -1);
   539→    const cx = origin.x + aimDir.x * (range / 2);
   540→    const cy = origin.y - 6 + aimDir.y * (range / 2);
   541→    const bodies = this.physics.bodiesInCircle(cx, cy, range / 2);
   542→    for (const body of bodies) {
   543→      const go = (body as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   544→      if (!go) continue;
   545→      const entityType = go.getData('entityType') as string | undefined;
   546→      const entity = go.getData('entity') as { id?: string; takeDamage?: (n: number) => void } | undefined;
   547→      if (!entity || !entityType) continue;
   548→      if (entityType === 'enemy' || entityType === 'boss') {
   549→        this.combat.dealDamage({
   550→          amount: this.stats.meleeDamage, type: 'melee', source: this.id,
   551→          target: entity.id ?? 'unknown',
   552→          knockback: { x: aimDir.x * 0.18, y: aimDir.y * 0.05 },
   553→          point: { x: (go as unknown as Phaser.GameObjects.Components.Transform).x, y: (go as unknown as Phaser.GameObjects.Components.Transform).y },
   554→        });
   555→      }
   556→    }
   557→  }
   558→
   559→  private updateAnimation(deltaMs: number): void {
   560→    this.animTime += deltaMs;
   561→    const pos = this.position;
   562→    const facing = this.facing === 'right' ? 1 : -1;
   563→    const body = this.sprite.body;
   564→    const vx = body?.velocity.x ?? 0;
   565→    const vy = body?.velocity.y ?? 0;
   566→    const isMoving = Math.abs(vx) > 0.5 && this.grounded;
   567→    const isJumping = !this.grounded;
   568→
   569→    const bob = isMoving ? Math.sin(this.animTime / 80) * 2 : 0;
   570→    if (this.mechaTorso) { this.mechaTorso.setPosition(pos.x, pos.y - 6 + bob); this.mechaTorso.setScale(facing, 1); }
   571→    if (this.core) {
   572→      this.core.setPosition(pos.x, pos.y - 6 + bob);
   573→      const pulse = 0.7 + Math.sin(this.animTime / 200) * 0.2;
   574→      this.core.setAlpha(pulse); this.core.setRadius(3 + pulse * 2);
   575→    }
   576→    if (this.mechaHead) this.mechaHead.setPosition(pos.x + facing * 6, pos.y - 18 + bob);
   577→    if (this.visor) { this.visor.setPosition(pos.x + facing * 6, pos.y - 19 + bob); this.visor.setAlpha(0.5 + Math.sin(this.animTime / 150) * 0.3); }
   578→    if (this.mechaLegL && this.mechaLegR) {
   579→      if (isJumping) {
   580→        this.mechaLegL.setPosition(pos.x - 6, pos.y + 10); this.mechaLegL.setAngle(-20);
   581→        this.mechaLegR.setPosition(pos.x + 6, pos.y + 10); this.mechaLegR.setAngle(20);
   582→      } else if (isMoving) {
   583→        const phase = this.animTime / 100;
   584→        this.mechaLegL.setPosition(pos.x - 8, pos.y + 14); this.mechaLegL.setAngle(Math.sin(phase) * 12);
   585→        this.mechaLegR.setPosition(pos.x + 8, pos.y + 14); this.mechaLegR.setAngle(Math.sin(phase + Math.PI) * 12);
   586→      } else {
   587→        this.mechaLegL.setPosition(pos.x - 8, pos.y + 14); this.mechaLegL.setAngle(0);
   588→        this.mechaLegR.setPosition(pos.x + 8, pos.y + 14); this.mechaLegR.setAngle(0);
   589→      }
   590→    }
   591→    if (this.gunArm) {
   592→      this.gunArm.setPosition(pos.x, pos.y - 6 + bob);
   593→      const dir = this.getAimDirection();
   594→      const targetAngle = Math.atan2(dir.y, dir.x);
   595→      this.aimAngle = Phaser.Math.Angle.RotateTo(this.aimAngle, targetAngle, 0.3);
   596→      this.gunArm.setRotation(this.aimAngle);
   597→      this.gunArm.setStrokeStyle(1, this.weaponData.color, 0.8);
   598→    }
   599→    // Landing impact
   600→    if (this.grounded && !this.wasGrounded && vy > 3) {
   601→      this.particles.dust(pos.x, pos.y + 20);
   602→      this.scene.cameras.main.shake(60, 0.003 * Math.min(vy / 5, 1));
   603→    }
   604→    this.wasGrounded = this.grounded;
   605→    // Flash while invulnerable
   606→    if (this.flashTimer > 0) {
   607→      this.flashTimer -= deltaMs;
   608→      const on = Math.floor(this.flashTimer / 80) % 2 === 0;
   609→      this.sprite.setAlpha(on ? 0.45 : 0);
   610→    } else {
   611→      this.sprite.setAlpha(0);
   612→    }
   613→  }
   614→
   615→  destroy(): void {
   616→    // Clear input callbacks so destroyed player doesn't receive input
   617→    InputSystem.clearCallbacks();
   618→    this.gunArm?.destroy(); this.mechaTorso?.destroy(); this.mechaHead?.destroy();
   619→    this.mechaLegL?.destroy(); this.mechaLegR?.destroy();
   620→    this.core?.destroy(); this.visor?.destroy();
   621→    if (this.sprite && this.sprite.active) this.sprite.destroy();
   622→  }
   623→}
