   171→  // ================ STATE MACHINE ================
   172→
   173→  private setState(next: GameState): void {
   174→    // Cleanup previous state
   175→    this.cleanupState();
   176→    this.state = next;
   177→    this.menuButtons = [];
   178→    this.menuFocusIndex = 0;
   179→    this.stateContainer = this.add.container(0, 0).setDepth(50);
   180→    switch (next) {
   181→      case 'menu': this.buildMenu(); break;
   182→      case 'hub': this.buildHub(); break;
   183→      case 'play': this.buildPlay(); break;
   184→      case 'gameover': this.buildGameOver(); break;
   185→      case 'victory': this.buildVictory(); break;
   186→    }
   187→    // setScrollFactor(0,0,true) AFTER all children are added by build* methods
   188→    this.stateContainer.setScrollFactor(0, 0, true);
   189→  }
   190→
   191→  private cleanupState(): void {
   192→    // Close any open overlays first
   193→    OverlayManager.closeAll();
   194→    // Destroy HUD if leaving play (hub is a separate environment — no HUD)
   195→    if (this.state === 'play') {
   196→      this.cleanupPlay();
   197→    }
   198→    if (this.menuNavHandler) {
   199→      window.removeEventListener('keydown', this.menuNavHandler);
   200→      this.menuNavHandler = null;
   201→    }
   202→    if (this.stateContainer) {
   203→      this.stateContainer.destroy(true);
   204→      this.stateContainer = null;
   205→    }
   206→    // Hide pause menu if visible
   207→    if (this.pauseMenuUI?.isVisible) this.pauseMenuUI.hide();
   208→    this.paused = false;
   209→  }

   211→  // ================ OVERLAY MANAGEMENT ================
   212→
   213→  /**
   214→   * Open an overlay from the current context.
   215→   * - From hub: overlay sits on top of hub. Closing returns to hub.
   216→   * - From play (paused): hide pause menu first. Closing reopens pause menu.
   217→   */
   218→  private openOverlay(id: OverlayId): void {
   219→    const parent: OverlayParent = this.state === 'hub' ? 'hub' : this.state === 'play' ? 'play' : 'menu';
   220→
   221→    // If from play, hide pause menu (paused stays true so play doesn't update)
   222→    if (parent === 'play') {
   223→      this.pauseMenuUI.hide();
   224→    }
   225→
   226→    // Build the overlay UI on demand
   227→    let ui: OverlayUI | null = null;
   228→    switch (id) {
   229→      case 'settings':
   230→        ui = new SettingsUI(this, () => this.closeOverlay());
   231→        break;
   232→      case 'skills':
   233→        ui = new SkillTreeUI(this, () => this.closeOverlay());
   234→        break;
   235→      case 'inventory':
   236→        ui = new InventoryUI(this, () => this.closeOverlay());
   237→        break;
   238→      case 'quests':
   239→        ui = new QuestUI(this, () => this.closeOverlay());
   240→        break;
   241→      case 'map':
   242→        ui = new WorldMapUI(this,
   243→          () => this.closeOverlay(),
   244→          (areaId: string) => this.fastTravel(areaId),
   245→        );
   246→        break;
   247→    }
   248→    if (!ui) return;
   249→    OverlayManager.open(id, ui, parent);
   250→  }
   251→
   252→  /** Close the current overlay and return to its parent. */
   253→  private closeOverlay(): void {
   254→    OverlayManager.close((parent) => {
   255→      if (parent === 'play') {
   256→        // Reopen pause menu
   257→        this.pauseMenuUI.show();
   258→      }
   259→      // If parent === 'hub', hub is still visible underneath — nothing to do
   260→    });
   261→  }
   262→
   263→  // ================ UPDATE LOOP ================
   264→
   265→  update(_time: number, deltaMs: number): void {
   266→    InputSystem.update();
   267→    const input = InputSystem.getState();
   268→
   269→    // *** Overlay input has highest priority — B/ESC closes, gamepad navigates ***
   270→    if (OverlayManager.hasOpen) {
   271→      OverlayManager.handleInput((parent) => {
   272→        if (parent === 'play') {
   273→          this.pauseMenuUI.show();
   274→        }
   275→      });
   276→      this.handleDialogueInput(input);
   277→      return;  // Block all other input while overlay is open
   278→    }
   279→
   280→    if (this.state === 'play') {
   281→      // ESC / Start = toggle pause
   282→      if (input.pausePressed) {
   283→        this.togglePause();
   284→      }
   285→      if (!this.paused) {
   286→        if (input.interactPressed) this.tryInteract();
   287→        // Freeze game while lore panel is open
   288→        if (!this.lorePanel) {
   289→          this.updatePlay(deltaMs);
   290→        }
   291→      } else {
   292→        // Paused — handle pause menu navigation
   293→        this.pauseMenuUI.handleNavigation();
   294→      }
   295→    } else if (this.state === 'menu' || this.state === 'hub' || this.state === 'gameover' || this.state === 'victory') {
   296→      // Gamepad + keyboard navigation
   297→      this.handleMenuGamepadNav(input);
   298→      // ESC in hub = back to menu
   299→      if (this.state === 'hub' && input.pausePressed) {
   300→        this.setState('menu');
   301→      }
   302→    }
   303→
   304→    this.handleDialogueInput(input);
   305→  }
   306→
   307→  private handleDialogueInput(input: import('../../systems/InputSystem').InputState): void {
   308→    if (this.dialogueUI?.isVisible) {
   309→      if (input.jumpPressed || input.firePressed) {
   310→        this.dialogueUI.advance();
   311→      }
   312→    }
   313→  }

   317→  private buildMenu(): void {
   318→    const c = this.stateContainer!;
   319→    const w = GAME.WIDTH, h = GAME.HEIGHT;
   320→
   321→    // === Background: starry night sky ===
   322→    const bg = this.add.graphics();
   323→    bg.setDepth(0);
   324→    bg.fillStyle(0x040814, 1);
   325→    bg.fillRect(0, 0, w, h);
   326→    for (let r = 500; r > 0; r -= 30) {
   327→      bg.fillStyle(0x0a1228, 0.02);
   328→      bg.fillCircle(w / 2, h * 0.35, r);
   329→    }
   330→    c.add(bg);
   331→
   332→    // Stars — 120 twinkling, drifting dots
   333→    for (let i = 0; i < 120; i++) {
   334→      const sx = Math.random() * w;
   335→      const sy = Math.random() * h * 0.75;
   336→      const size = 0.5 + Math.random() * 2;
   337→      const brightness = 0.2 + Math.random() * 0.8;
   338→      const starColor = Math.random() < 0.15 ? 0xffe0a0 : Math.random() < 0.3 ? 0xa0c0ff : 0xc0e0ff;
   339→      const star = this.add.circle(sx, sy, size, starColor, brightness);
   340→      star.setDepth(1);
   341→      c.add(star);
   342→      this.tweens.add({
   343→        targets: star,
   344→        alpha: { from: brightness * 0.1, to: brightness },
   345→        scale: { from: size * 0.5, to: size * 1.2 },
   346→        duration: 600 + Math.random() * 2500,
   347→        yoyo: true, repeat: -1,
   348→        delay: Math.random() * 3000,
   349→        ease: 'Sine.inOut',
   350→      });
   351→      this.tweens.add({
   352→        targets: star,
   353→        x: sx + (Math.random() - 0.5) * 30,
   354→        y: sy + (Math.random() - 0.5) * 20,
   355→        duration: 5000 + Math.random() * 8000,
   356→        yoyo: true, repeat: -1,
   357→        ease: 'Sine.inOut',
   358→      });
   359→    }
   360→
   361→    // Shooting stars
   362→    const shootingStarFunc = () => {
   363→      const ss = this.add.rectangle(0, 0, 40 + Math.random() * 30, 1.5, 0xffffff, 0.9);
   364→      ss.setBlendMode(Phaser.BlendModes.ADD);
   365→      ss.setDepth(1);
   366→      ss.setOrigin(0, 0.5);
   367→      ss.setRotation(0.3 + Math.random() * 0.2);
   368→      c.add(ss);
   369→      const startX = Math.random() * w * 0.7;
   370→      const startY = Math.random() * h * 0.3;
   371→      const endX = startX + 300 + Math.random() * 200;
   372→      const endY = startY + 120 + Math.random() * 80;
   373→      this.tweens.add({
   374→        targets: ss,
   375→        x: endX, y: endY,
   376→        alpha: { from: 0.9, to: 0 },
   377→        duration: 600 + Math.random() * 400,
   378→        onComplete: () => ss.destroy(),
   379→      });
   380→    };
   381→    this.time.addEvent({
   382→      delay: 3000, loop: true,
   383→      callback: () => { if (Math.random() < 0.4) shootingStarFunc(); },
   384→    });
   385→
   386→    // Brighter "beacon" stars with pulsing glow
   387→    for (let i = 0; i < 8; i++) {
   388→      const bx = Math.random() * w;
   389→      const by = Math.random() * h * 0.55;
   390→      const beaconColor = [0xffffff, 0x80a0ff, 0xffd0a0, 0xa0ffff][Math.floor(Math.random() * 4)];
   391→      const beacon = this.add.circle(bx, by, 2 + Math.random(), beaconColor, 1);
   392→      beacon.setDepth(1); beacon.setBlendMode(Phaser.BlendModes.ADD);
   393→      c.add(beacon);
   394→      const beaconGlow = this.add.circle(bx, by, 10 + Math.random() * 6, beaconColor, 0.12);
   395→      beaconGlow.setDepth(1); beaconGlow.setBlendMode(Phaser.BlendModes.ADD);
   396→      c.add(beaconGlow);
   397→      this.tweens.add({
   398→        targets: beaconGlow,
   399→        alpha: { from: 0.05, to: 0.25 },
   400→        scale: { from: 0.7, to: 1.3 },
   401→        duration: 1200 + Math.random() * 2000,
   402→        yoyo: true, repeat: -1, ease: 'Sine.inOut',
   403→      });
   404→      this.tweens.add({
   405→        targets: beacon,
   406→        alpha: { from: 0.6, to: 1 },
   407→        duration: 800 + Math.random() * 1200,
   408→        yoyo: true, repeat: -1,
   409→      });
   410→    }
   411→
   412→    // === Title: MECHA (very large) ===
   413→    const titleY = h * 0.3;
   414→    const glow = this.add.circle(w / 2, titleY, 250, 0x39d0d8, 0.05);
   415→    glow.setBlendMode(Phaser.BlendModes.ADD); glow.setDepth(2);
   416→    c.add(glow);
   417→    this.tweens.add({ targets: glow, alpha: { from: 0.03, to: 0.08 }, duration: 3000, yoyo: true, repeat: -1 });
   418→
   419→    const mechaText = this.add.text(w / 2, titleY, 'MECHA', {
   420→      fontFamily: 'monospace', fontSize: '96px', color: '#39d0d8',
   421→      stroke: '#000', strokeThickness: 8,
   422→    }).setOrigin(0.5).setDepth(3);
   423→    c.add(mechaText);
   424→
   425→    const protocolText = this.add.text(w / 2, titleY + 65, 'LAST PROTOCOL', {
   426→      fontFamily: 'monospace', fontSize: '22px', color: '#e0e8f0',
   427→      stroke: '#000', strokeThickness: 3, letterSpacing: 4,
   428→    }).setOrigin(0.5).setDepth(3);
   429→    c.add(protocolText);
   430→    this.tweens.add({ targets: protocolText, alpha: { from: 0.6, to: 1 }, duration: 2500, yoyo: true, repeat: -1 });
   431→
   432→    // === Small minimal buttons ===
   433→    const btnY = h * 0.6;
   434→    const btnGap = 48;
   435→    this.makeMenuBtn(w / 2, btnY, t('menu.start'), () => { AudioSystem.play('uiClick'); this.setState('hub'); });

   437→      AudioSystem.play('uiClick');

   439→        CheckpointSystem.init();
   440→        WorldSystem.initFromSave();
   441→        this.setState('hub');
   442→      }
   443→    }, !SaveSystem.hasCheckpoint());
   444→    this.makeMenuBtn(w / 2, btnY + btnGap * 2, t('menu.settings'), () => { AudioSystem.play('uiClick'); this.openOverlay('settings'); });
   445→    this.makeMenuBtn(w / 2, btnY + btnGap * 3, t('menu.how_to_play'), () => { AudioSystem.play('uiClick'); this.showHowToPlay(); });
   446→
   447→    // === Footer ===
   448→    c.add(this.add.text(w / 2, h - 25, t('game.version') + '  ·  PHASER 4.2 · MATTER.JS', {
   449→      fontFamily: 'monospace', fontSize: '9px', color: '#0a1220',
   450→    }).setOrigin(0.5).setDepth(3));
   451→
   452→    this.setupMenuNav();
   453→  }

   690→  private onCollisionStart = (event: MatterJS.IEventCollision<MatterJS.Body>): void => {
   691→    for (const pair of event.pairs) {
   692→      const aGo = (pair.bodyA as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   693→      const bGo = (pair.bodyB as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   694→      if (!aGo || !bGo) continue;
   695→      const aIsPlayer = aGo.getData('entityType') === 'player';
   696→      const bIsPlayer = bGo.getData('entityType') === 'player';
   697→      const aSection = aGo.getData('sectionId') as number | undefined;
   698→      const bSection = bGo.getData('sectionId') as number | undefined;
   699→      if (aIsPlayer && bSection) { this.enterSection(bSection); }
   700→      else if (bIsPlayer && aSection) { this.enterSection(aSection); }
   701→      else if (aIsPlayer && bGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   702→      else if (bIsPlayer && aGo.getData('isCheckpoint')) { this.activateCheckpoint(); }
   703→      else if (aIsPlayer && bGo.getData('isBossEntry')) { this.enterBossArena(); }
   704→      else if (bIsPlayer && aGo.getData('isBossEntry')) { this.enterBossArena(); }
   705→      if (aIsPlayer && bGo.getData('entityType') === 'enemy') { this.handleEnemyContact(bGo); }
   706→      else if (bIsPlayer && aGo.getData('entityType') === 'enemy') { this.handleEnemyContact(aGo); }
   707→      if (aIsPlayer && bGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   708→      else if (bIsPlayer && aGo.getData('entityType') === 'boss' && this.boss) { this.player.takeDamage(this.boss.getContactDamage()); }
   709→      // Hazard collision (spikes, lava, etc.)
   710→      if (aIsPlayer && bGo.getData('hazardDamage')) { this.handleHazard(bGo); }
   711→      else if (bIsPlayer && aGo.getData('hazardDamage')) { this.handleHazard(aGo); }
   712→    }
   713→  };

   888→  private updatePlay(deltaMs: number): void {
   889→    if (!this.player) return;
   890→    this.player.update(deltaMs);
   891→    this.render.update(this.time.now);
   892→    this.hud?.update();
   893→    // Ambient dust motes — atmospheric particles around player (per particles skill)
   894→    if (this.time.now % 200 < 16) {
   895→      this.particles.ambientDust(this.player.sprite.x, this.player.sprite.y - 40, 2);
   896→    }
   897→    // Projectiles
   898→    for (let i = this.projectiles.length - 1; i >= 0; i--) {
   899→      this.projectiles[i].update();
   900→      if (!this.projectiles[i].isAlive) this.projectiles.splice(i, 1);
   901→    }
   902→    // Enemies
   903→    const playerPos = this.player.position;
   904→    for (let i = this.enemies.length - 1; i >= 0; i--) {
   905→      const e = this.enemies[i];
   906→      if (!e.isAlive || !e.sprite || !e.sprite.active) { this.enemies.splice(i, 1); continue; }
   907→      try { e.update(deltaMs, playerPos); } catch { this.enemies.splice(i, 1); continue; }
   908→    }
   909→    // Boss
   910→    if (this.boss && this.boss.isAlive && this.boss.sprite && this.boss.sprite.active) {
   911→      try { this.boss.update(deltaMs); } catch { /* */ }
   912→      this.updateBossHealthBar();
   913→    }
   914→    // Out of bounds
   915→    if (this.player.sprite.y > GAME.HEIGHT + 80) {
   916→      this.player.takeDamage(25);
   917→      const area = WorldSystem.getCurrentArea();
   918→      if (area) {
   919→        const sec = area.sections.find(s => s.id === this.currentSection);

  1157→  private handleMenuGamepadNav(input: import('../../systems/InputSystem').InputState): void {
  1158→    if (this.menuButtons.length === 0) return;
  1159→    this.menuNavCooldown -= 16;

  1164→      this.updateMenuFocus(); AudioSystem.play('uiHover');
  1165→      this.menuNavCooldown = 110;  // freer — was 180

  1168→      this.updateMenuFocus(); AudioSystem.play('uiHover');
  1169→      this.menuNavCooldown = 110;
  1170→    }
  1171→    if (input.jumpPressed || input.firePressed) {
  1172→      AudioSystem.play('uiClick');
  1173→      const btn = this.menuButtons[this.menuFocusIndex];
  1174→      if (btn) btn.onSelect();
  1175→      this.menuNavCooldown = 300;
  1176→    }
  1177→  }

  1233→  private setupMenuNav(): void {
  1234→    this.menuNavHandler = (e: KeyboardEvent) => {
  1235→      if (this.menuButtons.length === 0) return;
  1236→      if (e.code === 'ArrowUp' || e.code === 'KeyW') {
  1237→        this.menuFocusIndex = (this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length;
  1238→        this.updateMenuFocus(); AudioSystem.play('uiHover');
  1239→      } else if (e.code === 'ArrowDown' || e.code === 'KeyS') {
  1240→        this.menuFocusIndex = (this.menuFocusIndex + 1) % this.menuButtons.length;
  1241→        this.updateMenuFocus(); AudioSystem.play('uiHover');
  1242→      } else if (e.code === 'Enter' || e.code === 'Space') {
  1243→        AudioSystem.play('uiClick');
  1244→        this.menuButtons[this.menuFocusIndex]?.onSelect();
  1245→      }
  1246→    };
  1247→    window.addEventListener('keydown', this.menuNavHandler);
  1248→    this.updateMenuFocus();
  1249→  }

  1255→      try {
  1256→        if (i === this.menuFocusIndex) {
  1257→          btn.bg.setFillStyle(0x0d1820, 1);
  1258→          btn.bg.setStrokeStyle(2, 0x39d0d8, 0.9);
  1259→          btn.bg.setScale(1.05);
  1260→          btn.text.setColor('#66f0ff');
  1261→        } else {
  1262→          btn.bg.setFillStyle(0x0a1018, 0.9);
  1263→          btn.bg.setStrokeStyle(1, 0x1a3040, 0.8);
  1264→          btn.bg.setScale(1);
  1265→          btn.text.setColor('#5a6470');
  1266→        }
  1267→      } catch { /* text canvas not ready — skip */ }
  1268→    });
  1269→  }

  1273→  private scheduleDelayed(delay: number, callback: () => void): Phaser.Time.TimerEvent {
  1274→    const timer = this.time.delayedCall(delay, () => {
  1275→      const idx = this.sequenceTimers.indexOf(timer);
  1276→      if (idx >= 0) this.sequenceTimers.splice(idx, 1);
  1277→      callback();
  1278→    });
  1279→    this.sequenceTimers.push(timer);
  1280→    return timer;
  1281→  }
