   395тЖТ   395тЖТ  private updatePlay(deltaMs: number): void {
   396тЖТ   396тЖТ    if (!this.player) return;
   397тЖТ   397тЖТ    this.player.update(deltaMs);
   398тЖТ   398тЖТ    this.graphics.update(this.time.now, deltaMs);
   399тЖТ   399тЖТ    this.hud?.update();
   400тЖТ   400тЖТ    this.bossBar?.update();
   401тЖТ   401тЖТ    // Projectiles
   402тЖТ   402тЖТ    for (let i = this.projectiles.length - 1; i >= 0; i--) {
   403тЖТ   403тЖТ      const p = this.projectiles[i];
   404тЖТ   404тЖТ      p.update(this);
   405тЖТ   405тЖТ      if (!p.isAlive) this.projectiles.splice(i, 1);
   406тЖТ   406тЖТ    }
   407тЖТ   407тЖТ    // Enemies
   408тЖТ   408тЖТ    const playerPos = this.player.position;
   409тЖТ   409тЖТ    const cx = this.cameras.main.scrollX + GAME.WIDTH / 2;
   410тЖТ   410тЖТ    for (let i = this.enemies.length - 1; i >= 0; i--) {
   411тЖТ   411тЖТ      const e = this.enemies[i];
   412тЖТ   412тЖТ      if (!e.isAlive || !e.sprite || !e.sprite.active) { this.enemies.splice(i, 1); continue; }
   413тЖТ   413тЖТ      if (Math.abs(e.sprite.x - cx) > GAME.WIDTH * 1.2) continue;
   414тЖТ   414тЖТ      try { e.update(deltaMs, playerPos); } catch { this.enemies.splice(i, 1); continue; }
   415тЖТ   415тЖТ      if (!e.isAlive) this.enemies.splice(i, 1);
   416тЖТ   416тЖТ    }
   417тЖТ   417тЖТ    // Boss
   418тЖТ   418тЖТ    if (this.boss && this.boss.isAlive && this.boss.sprite && this.boss.sprite.active) {
   419тЖТ   419тЖТ      try { this.boss.update(deltaMs); } catch { /* */ }
   420тЖТ   420тЖТ    }
   421тЖТ   421тЖТ    this.parallax?.update(deltaMs);
   422тЖТ   422тЖТ    this.atmosphere?.update(deltaMs);
   423тЖТ   423тЖТ    Ragdoll.update(this);
   424тЖТ   424тЖТ    // Out-of-bounds
   425тЖТ   425тЖТ    if (this.player.sprite.y > GAME.HEIGHT + 80) {
   426тЖТ   426тЖТ      this.player.takeDamage(25);
   427тЖТ   427тЖТ      this.respawnPlayerAtSection();
   428тЖТ   428тЖТ    }
   429тЖТ   429тЖТ    // Boss arena zoom
   430тЖТ   430тЖТ    if (this.bossArenaActive && this.boss && this.boss.isAlive) {
   431тЖТ   431тЖТ      this.cameras.main.setZoom(Phaser.Math.Linear(this.cameras.main.zoom, 0.85, 0.04));
   432тЖТ   432тЖТ    }
   433тЖТ   433тЖТ  }

   487тЖТ   487тЖТ  /**
   488тЖТ   488тЖТ   * Apply cinematic post-processing using Phaser 4.2's native Filter API.
   489тЖТ   489тЖТ   * Adds a Vignette (edge darkening) + ColorMatrix (contrast + slight warmth)
   490тЖТ   490тЖТ   * to the main camera. Idempotent тАФ only applies once per play session.
   491тЖТ   491тЖТ   */
   492тЖТ   492тЖТ  private applyCinematicGrade(): void {
   493тЖТ   493тЖТ    if (this.cinematicApplied) return;
   494тЖТ   494тЖТ    try {
   495тЖТ   495тЖТ      const cam = this.cameras.main;
   496тЖТ   496тЖТ      // eslint-disable-next-line @typescript-eslint/no-explicit-any
   497тЖТ   497тЖТ      const filters = (cam as unknown as { filters?: { external?: { addVignette?: (...args: unknown[]) => unknown }; internal?: { addColorMatrix?: () => { colorMatrix?: { contrast?: (v: number) => void; saturate?: (v: number) => void } } } } }).filters;
   498тЖТ   498тЖТ      if (!filters) return;
   499тЖТ   499тЖТ      // Vignette: centered, radius 0.5, strength 0.45, black, MULTIPLY blend
   500тЖТ   500тЖТ      // eslint-disable-next-line @typescript-eslint/no-explicit-any
   501тЖТ   501тЖТ      if (typeof filters.external?.addVignette === 'function') {
   502тЖТ   502тЖТ        filters.external.addVignette(0.5, 0.5, 0.5, 0.45, 0x000000);
   503тЖТ   503тЖТ      }
   504тЖТ   504тЖТ      // ColorMatrix: contrast boost for punchier visuals + slight saturation
   505тЖТ   505тЖТ      if (typeof filters.internal?.addColorMatrix === 'function') {
   506тЖТ   506тЖТ        const cmFilter = filters.internal.addColorMatrix();
   507тЖТ   507тЖТ        if (cmFilter?.colorMatrix) {
   508тЖТ   508тЖТ          if (typeof cmFilter.colorMatrix.contrast === 'function') cmFilter.colorMatrix.contrast(0.12);
   509тЖТ   509тЖТ          if (typeof cmFilter.colorMatrix.saturate === 'function') cmFilter.colorMatrix.saturate(1.08);
   510тЖТ   510тЖТ        }
   511тЖТ   511тЖТ      }
   512тЖТ   512тЖТ      this.cinematicApplied = true;
   513тЖТ   513тЖТ    } catch {
   514тЖТ   514тЖТ      // Filter API unavailable тАФ skip cinematic grade silently.
   515тЖТ   515тЖТ    }
   516тЖТ   516тЖТ  }

   518тЖТ   518тЖТ  /** Remove cinematic filters from the main camera (called on cleanup). */
   519тЖТ   519тЖТ  private removeCinematicGrade(): void {
   520тЖТ   520тЖТ    if (!this.cinematicApplied) return;
   521тЖТ   521тЖТ    try {
   522тЖТ   522тЖТ      const cam = this.cameras.main;
   523тЖТ   523тЖТ      // eslint-disable-next-line @typescript-eslint/no-explicit-any
   524тЖТ   524тЖТ      const filters = (cam as unknown as { filters?: { external?: { clear?: () => void }; internal?: { clear?: () => void } } }).filters;
   525тЖТ   525тЖТ      filters?.external?.clear?.();
   526тЖТ   526тЖТ      filters?.internal?.clear?.();
   527тЖТ   527тЖТ    } catch { /* ignore */ }
   528тЖТ   528тЖТ    this.cinematicApplied = false;
   529тЖТ   529тЖТ  }

   803тЖТ   803тЖТ  private onCollisionStart = (event: MatterJS.Events.CollisionStartEvent) => {
   804тЖТ   804тЖТ    for (const pair of event.pairs) {
   805тЖТ   805тЖТ      const aGo = (pair.bodyA as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   806тЖТ   806тЖТ      const bGo = (pair.bodyB as unknown as { gameObject?: Phaser.GameObjects.GameObject }).gameObject;
   807тЖТ   807тЖТ      if (!aGo || !bGo) continue;
   808тЖТ   808тЖТ      const aIsPlayer = aGo.getData('entityType') === 'player';
   809тЖТ   809тЖТ      const bIsPlayer = bGo.getData('entityType') === 'player';
   810тЖТ   810тЖТ      const aSection = aGo.getData('sectionId') as number | undefined;
   811тЖТ   811тЖТ      const bSection = bGo.getData('sectionId') as number | undefined;
   812тЖТ   812тЖТ      if (aIsPlayer && bSection) this.enterSection(bSection);
   813тЖТ   813тЖТ      else if (bIsPlayer && aSection) this.enterSection(aSection);
   814тЖТ   814тЖТ      else if (aIsPlayer && bGo.getData('isCheckpoint')) this.activateCheckpoint();
   815тЖТ   815тЖТ      else if (bIsPlayer && aGo.getData('isCheckpoint')) this.activateCheckpoint();
   816тЖТ   816тЖТ      else if (aIsPlayer && bGo.getData('isBossEntry')) this.enterBossArena();
   817тЖТ   817тЖТ      else if (bIsPlayer && aGo.getData('isBossEntry')) this.enterBossArena();
   818тЖТ   818тЖТ      // Health pickup
   819тЖТ   819тЖТ      else if (aIsPlayer && bGo.getData('isHealthPickup')) this.collectHealthPickup(bGo);
   820тЖТ   820тЖТ      else if (bIsPlayer && aGo.getData('isHealthPickup')) this.collectHealthPickup(aGo);
   821тЖТ   821тЖТ      // Spike trap
   822тЖТ   822тЖТ      else if (aIsPlayer && bGo.getData('isSpikeTrap')) this.player.takeDamage(15);
   823тЖТ   823тЖТ      else if (bIsPlayer && aGo.getData('isSpikeTrap')) this.player.takeDamage(15);
   824тЖТ   824тЖТ      if (aIsPlayer && bGo.getData('entityType') === 'enemy') this.handleEnemyContact(bGo);
   825тЖТ   825тЖТ      else if (bIsPlayer && aGo.getData('entityType') === 'enemy') this.handleEnemyContact(aGo);
   826тЖТ   826тЖТ      if (aIsPlayer && bGo.getData('entityType') === 'boss' && this.boss) this.player.takeDamage(this.boss.getContactDamage());
   827тЖТ   827тЖТ      else if (bIsPlayer && aGo.getData('entityType') === 'boss' && this.boss) this.player.takeDamage(this.boss.getContactDamage());
   828тЖТ   828тЖТ    }
   829тЖТ   829тЖТ  };

  1511тЖТ  1511тЖТ  /** Navigate menu with keyboard arrows + Enter, or gamepad D-pad/stick + A. */
  1512тЖТ  1512тЖТ  private menuNavHandler: ((e: KeyboardEvent) => void) | null = null;
  1513тЖТ  1513тЖТ
  1514тЖТ  1514тЖТ  private updateMenuNavigation(deltaMs: number): void {
  1515тЖТ  1515тЖТ    if (this.menuButtons.length === 0) return;
  1516тЖТ  1516тЖТ    this.menuNavCooldown -= deltaMs;
  1517тЖТ  1517тЖТ
  1518тЖТ  1518тЖТ    // Ensure a button is always focused
  1519тЖТ  1519тЖТ    if (this.menuFocusIndex >= this.menuButtons.length) this.menuFocusIndex = 0;
  1520тЖТ  1520тЖТ    if (this.menuFocusIndex < 0) this.menuFocusIndex = 0;
  1521тЖТ  1521тЖТ
  1522тЖТ  1522тЖТ    // Set up window keyboard listener once per state
  1523тЖТ  1523тЖТ    if (!this.menuNavHandler) {
  1524тЖТ  1524тЖТ      this.menuNavHandler = (e: KeyboardEvent) => {
  1525тЖТ  1525тЖТ        if (this.state === 'play') return; // skip during play
  1526тЖТ  1526тЖТ        if (this.menuButtons.length === 0) return;
  1527тЖТ  1527тЖТ        switch (e.code) {
  1528тЖТ  1528тЖТ          case 'KeyW': case 'ArrowUp':
  1529тЖТ  1529тЖТ            this.focusButton((this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length);
  1530тЖТ  1530тЖТ            Effects.play('uiHover');
  1531тЖТ  1531тЖТ            break;
  1532тЖТ  1532тЖТ          case 'KeyS': case 'ArrowDown':
  1533тЖТ  1533тЖТ            this.focusButton((this.menuFocusIndex + 1) % this.menuButtons.length);
  1534тЖТ  1534тЖТ            Effects.play('uiHover');
  1535тЖТ  1535тЖТ            break;
  1536тЖТ  1536тЖТ          case 'Enter': case 'Space': {
  1537тЖТ  1537тЖТ            const btn = this.menuButtons[this.menuFocusIndex];
  1538тЖТ  1538тЖТ            if (btn) { Effects.play('uiClick'); btn.onClick(); }
  1539тЖТ  1539тЖТ            break;
  1540тЖТ  1540тЖТ          }
  1541тЖТ  1541тЖТ          case 'Escape':
  1542тЖТ  1542тЖТ            this.menuGoBack();
  1543тЖТ  1543тЖТ            break;
  1544тЖТ  1544тЖТ        }
  1545тЖТ  1545тЖТ      };
  1546тЖТ  1546тЖТ      window.addEventListener('keydown', this.menuNavHandler);
  1547тЖТ  1547тЖТ    }
  1548тЖТ  1548тЖТ
  1549тЖТ  1549тЖТ    // Gamepad: Left stick OR D-pad to navigate, A to activate, B to go back
  1550тЖТ  1550тЖТ    const gp = GamepadManager.getState();
  1551тЖТ  1551тЖТ    if (this.menuNavCooldown <= 0) {
  1552тЖТ  1552тЖТ      if (gp.leftStickY < -0.3) {
  1553тЖТ  1553тЖТ        this.focusButton((this.menuFocusIndex - 1 + this.menuButtons.length) % this.menuButtons.length);
  1554тЖТ  1554тЖТ        Effects.play('uiHover');
  1555тЖТ  1555тЖТ        this.menuNavCooldown = 200;
  1556тЖТ  1556тЖТ      } else if (gp.leftStickY > 0.3) {
  1557тЖТ  1557тЖТ        this.focusButton((this.menuFocusIndex + 1) % this.menuButtons.length);
  1558тЖТ  1558тЖТ        Effects.play('uiHover');
  1559тЖТ  1559тЖТ        this.menuNavCooldown = 200;
  1560тЖТ  1560тЖТ      }
  1561тЖТ  1561тЖТ    }
  1562тЖТ  1562тЖТ    if (gp.jumpPressed) {
  1563тЖТ  1563тЖТ      const btn = this.menuButtons[this.menuFocusIndex];
  1564тЖТ  1564тЖТ      if (btn) { Effects.play('uiClick'); btn.onClick(); }
  1565тЖТ  1565тЖТ    }
  1566тЖТ  1566тЖТ    if (gp.backPressed) {
  1567тЖТ  1567тЖТ      this.menuGoBack();
  1568тЖТ  1568тЖТ    }
  1569тЖТ  1569тЖТ  }
